017package sph.vaadin.data.validator;
018
019
020/**
021 * VideojsUrlValidator validates the given URL string for a {@link sph.vaadin.ui.videojs.QuestionWindow} Component.
022 *
023 * @author  Sami Holck
024 * @author  Juha Mäkilä
025 * @version v0.1
026 * @since   3.11.2013
027 *
028 */
029public class VideojsUrlValidator extends UrlValidator {
030
031        /**
032         * 
033         */
034        private static final long serialVersionUID = -7214704407965419958L;
035
036        /**
037         * Constructs a validator with the default error message.
038         */
039        public VideojsUrlValidator() {
040                super("URL {0} is invalid for Videojs Component");
041        }
042
043        /**
044         * Constructs a validator with the given error message.
045         * 
046         * @param errorMessage the message to be included in an InvalidValueException
047         *        (with "{0}" replaced by the value that failed validation).
048         */
049        public VideojsUrlValidator(String errorMessage) {
050                super(errorMessage);
051        }
052
053        @Override
054        protected boolean isValidValue(String value) {
055                if (super.isValidValue(value)) {
056                        if (value.matches("http://www.youtube.com/(.*)") ||
057                                        value.matches("http://vimeo.com/(.*)") ||
058                                        value.matches("http://www.dailymotion.com/(.*)") ||
059                                        value.matches("(.*).mp4") ||
060                                        value.matches("(.*).ogg") ||
061                                        value.matches("(.*).WebM")) {
062                                return true;
063                        } else {
064                                if (this.getErrorMessage() == null || this.getErrorMessage().equals("")) {
065                                        this.setErrorMessage("The Stream type is not supprted");
066                                }
067                                return false;
068                        }
069                } else {
070                        if (this.getErrorMessage() == null || this.getErrorMessage().equals("")) {
071                                this.setErrorMessage("URL {0} is invalid");
072                        }
073                        return false;
074                }
075        }
076}
