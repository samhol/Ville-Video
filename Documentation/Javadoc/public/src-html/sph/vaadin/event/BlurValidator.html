001package sph.vaadin.event;
002
003import com.vaadin.event.FieldEvents;
004import com.vaadin.event.FieldEvents.BlurEvent;
005import com.vaadin.ui.AbstractTextField;
006
007/**
008 * Fired when a Field receives keyboard focus.
009 * 
010 * TextSelector selects the text from an AbstractTextField type object when the
011 * object is focused
012 * 
013 * @author  Sami Holck
014 * @author  Juha Mäkilä
015 * 
016 */
017public class BlurValidator implements FieldEvents.BlurListener {
018
019        /**
020         * version control in a Serializable class
021         */
022        private static final long serialVersionUID = -6526848061882745023L;
023
024        @Override
025        public void blur(BlurEvent event) {
026                AbstractTextField field = ((AbstractTextField) event.getComponent());
027                try {
028                        field.validate();
029                } catch (Exception e) {
030                        // field.selectAll();
031                }
032        }
033
034}
