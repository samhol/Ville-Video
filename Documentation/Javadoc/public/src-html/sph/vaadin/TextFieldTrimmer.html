017package sph.vaadin;
018
019import com.vaadin.event.FieldEvents;
020import com.vaadin.event.FieldEvents.BlurEvent;
021import com.vaadin.ui.AbstractTextField;
022
023/**
024 * TextFieldTrimmer trims leading and trailing whitespaces from an <em>AbstractTextField</em> component.
025 * 
026 * <p><strong class="Red">NOTE:</strong> can be used as a blur listener or by static method
027 * 
028 * <p><strong>copyright &copy; Ville-Video 2013</strong></p>
029 *
030 * @author  <a href="mailto:sami.holck@gmail.com">Sami Holck</a>
031 * @author  Juha Mäkilä
032 * @version 0.1
033 * @since   15.11.2013
034 */
035public class TextFieldTrimmer implements FieldEvents.BlurListener {
036
037        private static final long serialVersionUID = -6526848061882745023L;
038
039        /**
040         * Trims leading and trailing whitespace omitted from an <em>AbstractTextField</em>'s
041         * value when the object is blurred.
042         *
043         * @param event the event fired.
044         * @see   com.vaadin.event.FieldEvents.BlurListener#blur(com.vaadin.event.FieldEvents.BlurEvent)
045         * @see  String#trim()
046         */
047        @Override
048        public void blur(BlurEvent event) {
049                AbstractTextField tf = ((AbstractTextField) event.getComponent());
050                TextFieldTrimmer.trim(tf);
051        }
052
053        /**
054         * Returns the given <em>AbstractTextField</em>, with leading and trailing whitespace omitted from its value.
055         * 
056         * @param tf the text field to trim.
057         * @return the same text field trimmed.
058         */
059        public static AbstractTextField trim(AbstractTextField tf) {
060                String newVal = tf.getValue().trim();
061                tf.setValue(newVal);
062                return tf;
063        }
064
065}
