017package sph.vaadin.ui.videojs;
018
019import sph.vaadin.TextFieldValidator;
020import sph.vaadin.TextSelector;
021import sph.vaadin.VideojsUrlValidator;
022import sph.vaadin.ui.ComponentFactory;
023import sph.vaadin.ui.SPH_Theme;
024
025import com.vaadin.server.Page;
026import com.vaadin.ui.AbstractTextField.TextChangeEventMode;
027import com.vaadin.ui.Alignment;
028import com.vaadin.ui.Button;
029import com.vaadin.ui.Button.ClickEvent;
030import com.vaadin.ui.HorizontalLayout;
031import com.vaadin.ui.Notification;
032import com.vaadin.ui.TextField;
033
034/**
035 * A controller for {@link sph.vaadin.ui.videojs.Videojs#setSource(String, String)} component.
036 * 
037 * <p>VideojsSourceLoader is an UI component where an user can set the source of
038 * the {@link sph.vaadin.ui.videojs.Videojs#setSource(String, String)} component.</p>
039 * 
040 * <p><strong>copyright &copy; Ville-Video 2013</strong></p>
041 * 
042 * @author  <a href="mailto:sami.holck@gmail.com">Sami Holck</a>
043 * @author Juha Mäkilä
044 * @version v0.1
045 * @since 15.10.2013
046 * 
047 */
048public class VideojsSourceLoader extends VideojsController {
049
050        /**
051         * Constant serialized ID used for compatibility.
052         */
053        private static final long serialVersionUID = -1571516620549983694L;
054        /**
055         * the composition root for the VideojsSourceLoader component
056         */
057        private final HorizontalLayout compositionRoot = new HorizontalLayout();
058        /**
059         * load button
060         */
061        private final Button loadBtn = ComponentFactory.createSmallBlueButton("Load", true, SPH_Theme.LOAD_ICON_16PX);
062        /**
063         * video stream URL field
064         */
065        private final TextField videoUrlField = ComponentFactory.createTextField("URL:", "The URL of the video", true);
066
067        /**
068         * @param player
069         */
070        public VideojsSourceLoader(Videojs player) {
071                super(player);
072        }
073
074        /**
075         * Sets the path to the video stream and the MIME type of the video stream
076         * 
077         * @param videoSrc the source path of the video file
078         * @param mimeType the MIME type of the video
079         * @see   sph.vaadin.ui.videojs.Videojs#setSource(String, String)
080         */
081        public void setVideoSource(final String videoSrc, final String mimeType) {
082                this.videoUrlField.setValue(videoSrc);
083                this.getPlayer().setSource(videoSrc, mimeType);
084        }
085
086        /**
087         * Builds the loader part of the Video component
088         */
089        private void doLayout() {
090                this.videoUrlField.setInputPrompt("The URL of the video");
091                this.videoUrlField.addValidator(new VideojsUrlValidator());
092                this.videoUrlField.setNullRepresentation("");
093                this.videoUrlField.setWidth("300px");
094                this.compositionRoot.setSpacing(true);
095                this.compositionRoot.addComponents(videoUrlField, loadBtn);
096                this.compositionRoot.setComponentAlignment(videoUrlField, Alignment.BOTTOM_CENTER);
097                this.compositionRoot.setComponentAlignment(loadBtn, Alignment.BOTTOM_CENTER);
098                this.compositionRoot.setMargin(true);
099                this.setCompositionRoot(compositionRoot);
100        }
101
102        /*
103         * (non-Javadoc)
104         * 
105         * @see sph.vaadin.ui.videojs.VideojsPlayerController#process()
106         */
107        @Override
108        public void setup() {
109                this.doLayout();
110                this.videoUrlField.setImmediate(true);
111                this.videoUrlField.addValueChangeListener(new TextFieldValidator());
112                this.videoUrlField.addFocusListener(new TextSelector());
113                this.videoUrlField.setTextChangeEventMode(TextChangeEventMode.TIMEOUT);
114                this.loadBtn.addClickListener(new Button.ClickListener() {
115
116                        private static final long serialVersionUID = 402515738995537894L;
117
118                        @Override
119                        public void buttonClick(ClickEvent event) {
120                                if (videoUrlField.isValid()) {
121                                        getPlayer().setSource(videoUrlField.getValue(), parseMIME());
122                                } else {
123                                        Notification n = new Notification("Loading failed!",
124                                                        "<strong>Video URL:</strong> is not recognized as a valid video stream!",
125                                                        Notification.Type.WARNING_MESSAGE, true);
126                                        n.show(Page.getCurrent());
127                                }
128                        }
129                });
130        }
131
132        /**
133         * Returns the MIME type of the video stream URL if recognized.
134         * 
135         * @return the MIME type of the video stream URL if recognized.
136         * @see    sph.vaadin.ui.videojs.Videojs#setSource(String, String)
137         */
138        private String parseMIME() {
139                String src = videoUrlField.getValue();
140                String lcaseVal = src.toLowerCase();
141                if (src.matches("http://www.youtube.com/(.*)")) {
142                        return Videojs.YOUTUBE_MIME;
143                }
144                if (src.matches("http://vimeo.com/(.*)")) {
145                        return Videojs.VIMEO_MIME;
146                }
147                /*if (src.matches("http://www.dailymotion.com/(.*)")) {
148                        return Videojs.DAILYMOTION_MIME;
149                }*/
150                if (lcaseVal.matches("(.*).mp4")) {
151                        return Videojs.MP4_MIME;
152                }
153                if (lcaseVal.matches("(.*).ogg") || lcaseVal.matches("(.*).ogv")) {
154                        return Videojs.OGG_MIME;
155                }
156                if (lcaseVal.matches("(.*).webm")) {
157                        return Videojs.WEBM_MIME;
158                }
159                return null;
160        }
161}
