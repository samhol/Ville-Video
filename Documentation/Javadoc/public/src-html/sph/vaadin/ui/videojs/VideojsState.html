017package sph.vaadin.ui.videojs;
018
019import com.vaadin.shared.ui.JavaScriptComponentState;
020
021/**
022 * <strong>VideojsState object</strong> is a shared state object for <strong>
023 * {@link VideojsState}</strong> component.
024 * 
025 * <p>The basic communication from a server-side VideojsPlayer to its the
026 * client-side widget counterpart is handled using a VideojsPlayerState. The
027 * VideojsPlayerState is serialized transparently. VideojsPlayerState should be
028 * considered read-only on the client-side, as VideojsPlayerState is not
029 * serialized back to the server-side.</p>
030 * 
031 * <p>Values in the Shared State and in RPC calls are converted between Java and
032 * JavaScript using the following conventions:</p>
033 * 
034 * <ul>
035 * <li>Primitive Java numbers (<em>byte, char, int, long, float, double</em>) and their
036 * boxed types (<em>Byte, Character, Integer, Long, Float, Double</em>) are represented
037 * by JavaScript numbers.</li>
038 * <li>The primitive Java <em>boolean</em> and the boxed <em>Boolean</em> are represented by
039 * JavaScript booleans.</li>
040 * <li>Java Strings are represented by JavaScript strings.</li>
041 * <li>List, Set and all arrays in Java are represented by JavaScript arrays.</li>
042 * <li>Map<string, ?=""> in Java is represented by JavaScript object with fields
043 * corresponding to the map keys.</string,></li>
044 * <li>Any other Java Map is represented by a JavaScript array containing two
045 * arrays, the first contains the keys and the second contains the values in the
046 * same order.</li>
047 * <li>A Java Bean is represented by a JavaScript object with fields
048 * corresponding to the bean's properties.</li>
049 * <li>A Java Connector is represented by a JavaScript string containing the
050 * connector's id.</li>
051 * <li>A plugable serialization mechanism is provided for types not described
052 * here. Please refer to the documentation for specific types for serialization
053 * information.</li>
054 * </ul>
055 * 
056 *
057 * <p><strong>copyright &copy; Ville-Video 2013</strong></p>
058 *
059 * @author  <a href="mailto:sami.holck@gmail.com">Sami Holck</a>
060 * @author  Juha Mäkilä
061 * @version 2.1
062 * @since   11.09.2013
063 */
064public class VideojsState extends JavaScriptComponentState {
065
066        /**
067         * version control in a Serializable class
068         */
069        private static final long serialVersionUID = 1L;
070        /**
071         * the unique id of the Video.js component
072         */
073        public String videoJsId;
074        /**
075         * the path to the video as an URL string
076         */
077        public String videoSrc;
078        /**
079         * the mime type of the video
080         */
081        public String mimetype;
082        /**
083         * <strong>{@link Videojs} server to Video.js client action flag:</strong>
084         * <p>
085         * Indicates that a player dimension change action is required from a server-side {@link Videojs}
086         * to its the client-side widget counterpart.
087         * </p>
088         */
089        /**
090         * the width of the Video.js component
091         */
092        public int vjsWidth;
093        /**
094         * the height of the Video.js component
095         */
096        public int vjsHeight;
097        /**
098         * Video.js component masking flag (true: video component is masked, false:
099         * video component is unmasked)
100         */
101        public boolean isMasked;
102        /**
103         * Video.js component mask label text
104         */
105        public String maskLabel;
106        /**
107         * Delay in milliseconds before Video.js component is masked
108         */
109        public int maskDelay;
110        /**
111         * Videojs component flag for video stream position seeking (true: seeking
112         * enabled, false: seeking disabled)
113         */
114        public boolean seekingEnabled;
115}
