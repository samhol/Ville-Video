174        /**
175         * Sets the initial properties of the Video.js player
176         */
177        protected void init() {
178                this.addFunction("videoEventFired", new JavaScriptFunction() {
179
180                        private static final long serialVersionUID = -4245967454493315292L;
181
182                        @Override
183                        public void call(JSONArray arguments) throws JSONException {
184                                try {
185                                        if (arguments.length() > 1) {
186                                                currentTime = arguments.getDouble(1);
187                                                String eventName = arguments.getString(0);
188                                                double triggerTime = arguments.getDouble(1);
189                                                //System.out.println(eventName + ":" + triggerTime);
190                                                if (eventName.equals(VjsListener.DURATIONCHANGE_EVENT) && arguments.length() > 2) {
191                                                        duration = arguments.getDouble(2);
192                                                        //System.out.println("duration: " + duration);
193                                                }
194                                                if (eventName.equals(VjsListener.PLAY_EVENT)) {
195                                                        isPaused = false;
196                                                }
197                                                if (eventName.equals(VjsListener.PAUSE_EVENT)) {
198                                                        isPaused = true;
199                                                }
200                                                videojsEventManager.callListeners(VjsListener.ANY_EVENT, Videojs.this, triggerTime);
201                                                videojsEventManager.callListeners(eventName, Videojs.this, triggerTime);
202                                        }
203                                } catch (JSONException e) {
204                                        // We'll log in the console, you might not want to
205                                        JavaScript.getCurrent().execute(
206                                                        "console.error('" + e.getMessage() + "')");
207                                        System.out.println(e.getMessage());
208                                }
209                        }
210                });
211                addDetachListener(new DetachListener() {
212
213                        private static final long serialVersionUID = 5583909485994600778L;
214
215                        @Override
216                        public void detach(DetachEvent event) {
217                                videojsEventManager.clear();
218                                JavaScript.getCurrent().execute("videojs('" + videoJsId +"').dispose();");
219                        }
220                });
221        }

352                this.pauseAt(times);
353        }
354
355        /**
356         * Pauses the video play back on the given times.
357         * 
358         * <p>
359         * <b>NOTE: </b> this method replaces all previously defined pause times.
360         * </p>
361         * 
362         * @param times Pausing times.
363         */
364        public <T extends Number> void pauseAt(Collection<T> times) {
365                this.pauseAts.clear();
366                if (times != null) {
367                        for(T time: times) {
368                                this.pauseAts.add(time.doubleValue());
369                        }
370                }
371                callFunction("pauseAt", this.pauseAts);
372        }
373
374        /**
375         * Removes an predefined pause time from the player.
376         * 
377         * @param time removed pausing time.
378         */
379        public <T extends Number> void removePauseAtTime(T time) {
380                if (this.pauseAts.remove(time.doubleValue())) {
381                        callFunction("pauseAt", this.pauseAts);
382                }
383        }

387         */

522        /**
523         * {@link sph.event.EventListener} interface for listening to client side Video.js Events
524         * delegated to and fired by a {@link Videojs} player component.
525         * 
526         * @author  <a href="mailto:sami.holck@gmail.com">Sami Holck</a>
527         * @author  Juha Mäkilä
528         * @version 2.01
529         * @since   11.10.2013
530         */
531        public interface VjsListener extends EventListener<String, Videojs, Double> {
532
533                /**
534                 * loadstart event (Fired when the user agent begins looking for media data)
535                 */
536                public static final String LOADSTART_EVENT = "loadstart";
537                /**
538                 * loadedmetadata event (Fired when the player has initial duration and dimension information)
539                 */
540                public static final String LOADEDMETADATA_EVENT = "loadedmetadata";
541                /**
542                 * loadeddata event (Fired when the player has downloaded data at the current playback position)
543                 */
544                public static final String LOADEDDATA_EVENT = "loadeddata";
545                /**
546                 * loadedalldata event (Fired when the player has finished downloading the source data)
547                 */
548                public static final String LOADEDALLDATA_EVENT = "loadedalldata";
549                /**
550                 * play event (Fired whenever the media begins or resumes playback)
551                 */
552                public static final String PLAY_EVENT = "play";
553                /**
554                 * pause event (Fired whenever the media has been paused)
555                 */
556                public static final String PAUSE_EVENT = "pause";
557                /**
558                 * timeupdate event (Fired when the current playback position has
559                 * changed. During playback this is fired every 15-250 milliseconds,
560                 * depending on the playback technology in use.)
561                 */
562                public static final String TIMEUPDATE = "timeupdate";
563                /**
564                 * ended event (Fired when the end of the media resource is reached)
565                 */
566                public static final String ENDED_EVENT = "ended";
567                /**
568                 * durationchange event (Fired when the duration of the media resource is changed, or known for the first time)
569                 */
570                public static final String DURATIONCHANGE_EVENT = "durationchange";
571                /**
572                 * progress event (Fired while the user agent is downloading media data)
573                 */
574                public static final String PROGRESS_EVENT = "progress";
575                /**
576                 * resize event (Fired when the width and/or height of the video window changes)
577                 */
578                public static final String RESIZE_EVENT = "resize";
579                /**
580                 * volumechange event (Fired when the volume changes)
581                 */
582                public static final String VOLUMECHANGE_EVENT = "volumechange";
583                /**
584                 * error event (Fired when there is an error in playback)
585                 */
586                public static final String ERROR_EVENT = "error";
587                /**
588                 * fullscreenchange event (Fired when the player switches in or out of fullscreen mode)
589                 */
590                public static final String FULLSCREENCHANGE_EVENT = "fullscreenchange";
591                /**
592                 * loadstart event (Fired when the user agent begins looking for media data)
593                 */
594                public static final String ANY_EVENT = "any";
595
596                /**
597                 * {@link Videojs} Component has triggered an event.
598                 * 
599                 * @param eventName name of the {@link Videojs.VjsListener VideojsListener} event.
600                 * @param source the source component of the {@link Videojs.VjsListener VideojsListener} event.
601                 * @param triggerTime the playback time of the videostream when the  event is triggered.
602                 */
603                @Override
604                public void on(String eventName, Videojs source, Double triggerTime);
605
606        }
