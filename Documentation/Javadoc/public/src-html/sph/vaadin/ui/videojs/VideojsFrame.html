017package sph.vaadin.ui.videojs;
018
019import com.vaadin.server.Sizeable;
020import com.vaadin.ui.Alignment;
021import com.vaadin.ui.CustomComponent;
022import com.vaadin.ui.HorizontalLayout;
023import com.vaadin.ui.VerticalLayout;
024
025/**
026 * A container for <b>{@link sph.vaadin.ui.videojs.Videojs}</b>
027 * component and its basic controllers <b>{@link sph.vaadin.ui.videojs.VideojsSourceLoader}</b>,
028 * <b>{@link sph.vaadin.ui.videojs.VideojsSeekToController}</b>
029 * and <b>{@link sph.vaadin.ui.videojs.VideojsTimeJumper}</b>.
030 * 
031 * <p><strong>copyright &copy; Ville-Video 2013</strong></p>
032 *
033 * @author  <a href="mailto:sami.holck@gmail.com">Sami Holck</a>
034 * @author  Juha Mäkilä
035 * @version 1.0
036 * @since   11.10.2013
037 * 
038 */
039public class VideojsFrame extends CustomComponent {
040
041        /**
042         * version control in a Serializable class
043         */
044        private static final long serialVersionUID = 1832295496823246699L;
045        /**
046         * the composition root for the VideoFrame component
047         */
048        private final VerticalLayout compositionRoot = new VerticalLayout();
049        /**
050         * the bottom container for the VideoFrame component
051         */
052        private final HorizontalLayout controlsLayout = new HorizontalLayout();
053        /**
054         * Video.js component
055         */
056        private final Videojs player = new Videojs();
057        /**
058         * the video source loader for the VideoFrame component
059         */
060        private final VideojsSourceLoader vjsSL;
061
062        private final VideojsTimeJumper vjsDC;
063        private final VideojsSeekToController vjsSTC;
064
065        /**
066         * 
067         */
068        public VideojsFrame() {
069                this("", "");
070        }
071
072        /**
073         * 
074         * @param videoSrc the source path of the video file
075         * @param mimeType the MIME type of the video
076         */
077        public VideojsFrame(final String videoSrc, final String mimeType) {
078                this.vjsSL = new VideojsSourceLoader(this.player);
079                this.vjsSL.setup();
080                this.vjsDC = new VideojsTimeJumper(this.player);
081                this.vjsDC.setup();
082                this.vjsSTC = new VideojsSeekToController(this.player);
083                this.vjsSTC.setup();
084                this.doLayout();
085                this.vjsSL.setVideoSource(videoSrc, mimeType);
086        }
087
088        /**
089         * Sets the path to the video stream and the MIME type of the video stream
090         * 
091         * @param videoSrc the source path of the video file
092         * @param mimeType the MIME type of the video
093         */
094        public void setVideoSource(final String videoSrc, final String mimeType) {
095                this.vjsSL.setVideoSource(videoSrc, mimeType);
096        }
097
098        /**
099         * Sets the visibility of the Loader component.
100         * 
101         * <p>
102         * Visible components are drawn in the user interface, while invisible ones
103         * are not. The effect is not merely a cosmetic CSS change - no information
104         * about an invisible component will be sent to the client. The effect is
105         * thus the same as removing the component from its parent.
106         * </p>
107         * 
108         * <p>
109         * A component is visible only if all of its parents are also visible. If a
110         * component is explicitly set to be invisible, changes in the visibility of
111         * its parents will not change the visibility of the component.
112         * </p>
113         * 
114         * @param visible the boolean value specifying if the loader component
115         *            should be visible after the call or not.
116         */
117        public void setLoaderVisible(boolean visible) {
118                this.vjsSL.setVisible(visible);
119        }
120
121        /**
122         * Sets the visibility of the VideoStreamControls component.
123         * 
124         * <p>
125         * Visible components are drawn in the user interface, while invisible ones
126         * are not. The effect is not merely a cosmetic CSS change - no information
127         * about an invisible component will be sent to the client. The effect is
128         * thus the same as removing the component from its parent.
129         * </p>
130         * 
131         * <p>
132         * A component is visible only if all of its parents are also visible. If a
133         * component is explicitly set to be invisible, changes in the visibility of
134         * its parents will not change the visibility of the component.
135         * </p>
136         * 
137         * @param visible the boolean value specifying if the VideoStreamControls
138         *            component should be visible after the call or not.
139         */
140        public void setVideoStreamControlsVisible(boolean visible) {
141                this.vjsDC.setVisible(visible);
142                this.vjsSTC.setVisible(visible);
143        }
144
145        /**
146         * <p>
147         * Enables or disables the Loader. The user can not interact disabled
148         * components, which are shown with a style that indicates the status,
149         * usually shaded in light gray color. Components are enabled by default.
150         * </p>
151         * 
152         * @param enabled a boolean value specifying if the Loader should be enabled
153         *            or not
154         */
155        public void setLoaderEnabled(boolean enabled) {
156                this.vjsSL.setEnabled(enabled);
157        }
158
159        /**
160         * <p>
161         * Enables or disables the component. The user can not interact disabled
162         * components, which are shown with a style that indicates the status,
163         * usually shaded in light gray color. Components are enabled by default.
164         * </p>
165         * 
166         * @param enabled a boolean value specifying if the component should be
167         *            enabled or not
168         */
169        public void setVideoStreamControlsEnabled(boolean enabled) {
170                this.vjsDC.setEnabled(enabled);
171                this.vjsSTC.setEnabled(enabled);
172        }
173
174        /**
175         * Returns the source path of the video stream.
176         * 
177         * @return the source path of the video stream.
178         */
179        public String getVideoSrc() {
180                return this.player.getVideoSrc();
181        }
182
183        /**
184         * Returns the MIME Type of the video stream.
185         * 
186         * @return the MIME type of the video stream.
187         */
188        public String getMimeType() {
189                return this.player.getMimeType();
190        }
191
192        private void doLayout() {
193                this.controlsLayout.setWidth(100, Sizeable.Unit.PERCENTAGE);
194                this.controlsLayout.setDefaultComponentAlignment(Alignment.BOTTOM_CENTER);
195                this.controlsLayout.addComponents(this.vjsSL, this.vjsSTC, this.vjsDC);
196                //this.compositionRoot.addComponent(this.controlsLayout);
197                this.compositionRoot.addComponents(this.controlsLayout, this.player);
198                this.setCompositionRoot(this.compositionRoot);
199        }
200
201        /**
202         * Returns the Video.js player component.
203         * 
204         * @return the Video.js player component.
205         */
206        public Videojs getVideojsPlayer() {
207                return this.player;
208        }
209}
