017package sph.vaadin.ui.svg;
018
019import java.util.ArrayList;
020
021import com.vaadin.shared.ui.JavaScriptComponentState;
022
023/**
024 * <strong>SvgDotsState</strong> object is a shared state object for <strong>
025 * {@link SvgDots}</strong> component.
026 * 
027 * <p>
028 * The basic communication from a server-side VideojsPlayer to its the
029 * client-side widget counterpart is handled using a VideojsPlayerState. The
030 * VideojsPlayerState is serialized transparently. VideojsPlayerState should be
031 * considered read-only on the client-side, as VideojsPlayerState is not
032 * serialized back to the server-side.
033 * </p>
034 * 
035 * <p>
036 * Values in the Shared State and in RPC calls are converted between Java and
037 * JavaScript using the following conventions:
038 * </p>
039 * 
040 * <ul>
041 * <li>Primitive Java numbers (byte, char, int, long, float, double) and their
042 * boxed types (Byte, Character, Integer, Long, Float, Double) are represented
043 * by JavaScript numbers.</li>
044 * <li>The primitive Java boolean and the boxed Boolean are represented by
045 * JavaScript booleans.</li>
046 * <li>Java Strings are represented by JavaScript strings.</li>
047 * <li>List, Set and all arrays in Java are represented by JavaScript arrays.</li>
048 * <li>Map<string, ?=""> in Java is represented by JavaScript object with fields
049 * corresponding to the map keys.</string,></li>
050 * <li>Any other Java Map is represented by a JavaScript array containing two
051 * arrays, the first contains the keys and the second contains the values in the
052 * same order.</li>
053 * <li>A Java Bean is represented by a JavaScript object with fields
054 * corresponding to the bean's properties.</li>
055 * <li>A Java Connector is represented by a JavaScript string containing the
056 * connector's id.</li>
057 * <li>A pluggable serialization mechanism is provided for types not described
058 * here. Please refer to the documentation for specific types for serialization
059 * information.</li>
060 * </ul>
061 * 
062 * <p><strong>copyright &copy; Ville-Video 2013</strong></p>
063 *
064 * @author  <a href="mailto:sami.holck@gmail.com">Sami Holck</a>
065 * @author  Juha Mäkilä
066 * @version 0.1
067 * @since   19.10.2013
068 *
069 */
070public class SvgDotsState extends JavaScriptComponentState {
071
072        private static final long serialVersionUID = -3111856093333159122L;
073
074        public ArrayList<SvgDots.Dot> dots;
075
076}
