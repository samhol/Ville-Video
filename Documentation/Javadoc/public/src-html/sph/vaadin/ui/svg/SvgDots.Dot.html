032/**
033 * SvgDots Component draws colored circles (dots) horizontally to an SVG canvas.

035 * <p>Uses <a href="http://raphaeljs.com/">Raphaël—JavaScript Library</a> for SVG graphics.</p>

052        private static final long serialVersionUID = 3689118911476002692L;

054        private ArrayList<SvgDots.Dot> dots = new ArrayList<SvgDots.Dot>();

056        /**
057         * an SvgDots EventManager
058         */
059        private final BasicEventManager<String, SvgDots, Integer, DotListener> dotEventManager = new BasicEventManager<String, SvgDots, Integer, DotListener>();

061        /**
062         * <strong>Default Constructor:</strong> Creates an empty SVG canvas for {@link SvgDots.Dot} elements.
063         */
064        public SvgDots() {
065                this(0);
066                this.init();
067        }
068
069        /**
070         * <strong>Constructor:</strong> Creates amount of <code>count</code>
071         * Silver grey colored clickable {@link SvgDots.Dot} elements into an SVG canvas.
072         * 
073         * @.pre  count >= 0
074         * @.post {@link #size()} == count & FORALL({@link SvgDots.Dot dot} : this};
075         *        {@link SvgDots.Dot#getColor() dot.getColor()}.{@link String#equals(Object) equals("")} &
076         *        {@link SvgDots.Dot#getTitle() dot.getTitle()}.{@link String#equals(Object) equals("")} &
077         *        {@link SvgDots.Dot#isClickable() dot.isClickable()} == true)
078         * @param count the number of the created {@link SvgDots.Dot} elements.
079         */
080        public SvgDots(int count) {
081                ArrayList<SvgDots.Dot> dotsData = new ArrayList<SvgDots.Dot>();
082                for (int i = 0; i < count; i++) {
083                        dotsData.add(new SvgDots.Dot("Silver"));
084                }
085                this.draw(dotsData);
086                this.init();
087        }
088
089        /**
090         * Draws an amount of <code>dots.size()</code> {@link SvgDots.Dot} elements to the SVG canvas.
091         * 
092         * @param dots the list of the SVG dots.
093         */
094        public SvgDots(final Collection<SvgDots.Dot> dots) {
095                this.draw(dots);
096                this.init();
097        }
098
099        /**
100         * Initializes the dots.
101         */
102        private void init() {
103                this.setId("svg_" + this.hashCode());
104                this.addFunction("delegateDotEvent", new JavaScriptFunction() {
105
106                        private static final long serialVersionUID = -4245967454493315292L;
107
108                        @Override
109                        public void call(JSONArray arguments) throws JSONException {
110                                if (arguments.length() > 1) {
111                                        dotEventManager.callListeners(arguments.getString(0), SvgDots.this, arguments.getInt(1));
112                                }
113                        }
114                });
115                this.addDetachListener(new DetachListener() {
116
117                        private static final long serialVersionUID = 5583909485994600778L;
118
119                        @Override
120                        public void detach(DetachEvent event) {
121                                dotEventManager.clear();
122                        }
123                });
124        }
125
126        /**
127         * Draws the given {@link SvgDots.Dot} elements to the SVG canvas.
128         * 
129         * <p><strong class="red">NOTE:</strong> removes all previously defined {@link SvgDots.Dot} elements.</p>
130         * 
131         * @param dotList a collection of dots.
132         */
133        public void draw(final Collection<SvgDots.Dot> dotList) {
134                ArrayList<SvgDots.Dot> list = new ArrayList<SvgDots.Dot>(dotList.size());
135                list.addAll(dotList);
136                this.dots = list;
137                getState().dots = list;
138        }
139
140        /**
141         * Returns the element at the specified position in this list.
142         * 
143         * @param   index index of the element to return.
144         * @return  index of the element to return.
145         * @throws  IndexOutOfBoundsException if the index is out of range (index < 0 || index >= size()).
146         */
147        public SvgDots.Dot getDot(int index) {
148                return this.dots.get(index);
149        }
150
151        /**
152         * Replaces the {@link SvgDots.Dot} element at the given index.
153         * 
154         * @param index the index of the replaced the {@link SvgDots.Dot} element.
155         * @param dot the {@link SvgDots.Dot} element.
156         * @throws IndexOutOfBoundsException - if the index is out of range (index < 0 || index >= size())
157         */
158        public void replaceDot(int index, SvgDots.Dot dot) {
159                this.dots.set(index, dot);
160                this.callFunction("replaceDot", index, dot);
161        }
162

167         * @param color the color of the {@link SvgDots.Dot}.
168         */
169        public void setDotColor(int index, String color) {
170                this.callFunction("setDotColor", index, color);
171        }
172
173        /**

176         * @param index the index of the {@link SvgDots.Dot}.
177         * @param title the color of the {@link SvgDots.Dot}.
178         */
179        public void setDotTitle(int index, String title) {
180                this.callFunction("setDotTitle", index, title);
181        }

237         * Dot contains all the required information about a single SVG dot in {@link SvgDots} component.

246        public static class Dot implements Serializable, Cloneable {
247
248                private static final long serialVersionUID = 5743971405439331162L;
249
250                /**
251                 * Property <code>title</code>: the title attribute of the SVG dot
252                 */
253                private String title = "";
254                /**
255                 * Property <code>color</code>: the color of the SVG dot
256                 */
257                private String color = "Silver";
258                /**
259                 * Property <code>clickable</code>: the color of the SVG dot
260                 */
261                private boolean clickable = true;
262                /**
263                 * Property <code>clickable</code>: the color of the SVG dot
264                 */
265                private Object data;
266
267                /**
268                 * <strong>Default Constructor:</strong> Creates an empty {@link SvgDots.Dot}
269                 * element for an {@link SvgDots} component.

271                public Dot() {
272                        this("Silver");
273                }
274
275                /**
276                 * <strong>Constructor:</strong> Creates an {@link SvgDots.Dot} element
277                 * of color <em>color</em> for an {@link SvgDots} component.

280                 */

282                        this(color, "");
283                }
284
285                /**
286                 * <strong>Constructor:</strong> Creates an {@link SvgDots.Dot} element

297                /**

299                 * of color <em>color</em> and title <em>title</em> for an {@link SvgDots} component.
300                 * 
301                 * @param color the color of the SVG dot.

303                 * @param clickable a boolean value specifying whether the dot is clickable or not.
304                 */

307                        this.color = color;

384                /**
385                 * Returns a hash code value for the object. This method is supported for
386                 * the benefit of hash tables such as those provided by {@link java.util.HashMap}.
387                 * 
388                 * @return a hash code value for this object.
389                 * @see java.lang.Object#hashCode()
390                 */
391                @Override
392                public int hashCode() {
393                        final int prime = 31;
394                        int result = super.hashCode();
395                        result = prime * result + (clickable ? 1231 : 1237);
396                        result = prime * result + ((color == null) ? 0 : color.hashCode());
397                        result = prime * result + ((title == null) ? 0 : title.hashCode());
398                        return result;
399                }

401                /**
402                 * Indicates whether some other object is equal to this {@link SvgDots.Dot} element.
403                 * 
404                 * @.post getColor().equals(RESULT.getColor()) &
405                 *        getTitle().equals(RESULT.getTitle()) &
406                 *        isClickable() == RESULT.isClickable()
407                 * @see java.lang.Object#equals(java.lang.Object)
408                 */
409                @Override
410                public boolean equals(Object obj) {
411                        if (this == obj) {
412                                return true;
413                        }
414                        if (obj == null) {
415                                return false;
416                        }
417                        if (getClass() != obj.getClass()) {
418                                return false;
419                        }
420                        Dot other = (Dot) obj;
421                        if (clickable != other.clickable) {
422                                return false;
423                        }
424                        if (color == null) {
425                                if (other.color != null) {
426                                        return false;
427                                }
428                        } else if (!color.equals(other.color)) {
429                                return false;
430                        }
431                        if (title == null) {
432                                if (other.title != null) {
433                                        return false;
434                                }
435                        } else if (!title.equals(other.title)) {
436                                return false;
437                        }
438                        return true;
439                }

441                /**

444                 * @return a string representation of this {@link SvgDots.Dot} element.
445                 * @see    java.lang.Object#toString()
446                 */

448                public String toString() {
449                        StringBuilder builder = new StringBuilder();
450                        builder.append("Dot [title=");
451                        builder.append(title);
452                        builder.append(", color=");
453                        builder.append(color);
454                        builder.append(", clickable=");
455                        builder.append(clickable);
456                        builder.append(", data=");
457                        builder.append(data);
458                        builder.append("]");
459                        return builder.toString();
460                }
461        }

467         * @param color the color to set
468         * @param title the title to set.
469         * @param clickable a boolean value specifying whether the dot is clickable or not.

473                Dot dot = new Dot();
474                dot.setColor(color);
475                dot.setTitle(title);
476                dot.setClickable(clickable);
477                return dot;

480        /**
481         * SvgDotsListener interface for listening to {@link SvgDots.Dot} events fired by an {@link SvgDots} component.
482         * 
483         * @author  <a href="mailto:sami.holck@gmail.com">Sami Holck</a>
484         * @author  Juha Mäkilä
485         * @version 1.0
486         * @since   11.10.2013
487         */
488        public static interface DotListener extends EventListener<String, SvgDots, Integer> {
489                /**
490                 * click event (Fired whenever a dot has been clicked)
491                 */
492                public static String CLICK_EVENT = "click";
493
494                /**
495                 * Called in response to a {@link SvgDots.Dot} event if the present listener is registered to listen for the given event.
496                 * 
497                 * @param eventName Name of the event.
498                 * @param canvas the source SVG canvas of the SvgDot Event.
499                 * @param dotIndex the index of the dot that triggered the event.
500                 */
501                @Override
502                public void on(String eventName, SvgDots canvas, Integer dotIndex);
503        }
