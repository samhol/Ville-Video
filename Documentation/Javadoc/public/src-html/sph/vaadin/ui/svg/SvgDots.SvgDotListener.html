020import java.util.ArrayList;

022
023import org.apache.commons.lang.builder.EqualsBuilder;

034/**
035 * SvgDots Component draws colored circles (dots) horizontally to an SVG canvas.
036 * 
037 * @author Sami Holck
038 * @author Juha Mäkilä
039 * @version 1.01
040 * @since 19.10.2013
041 * 
042 */
043@com.vaadin.annotations.JavaScript({ "javascript/raphael-min.js", "javascript/SvgDots.js" })
044public class SvgDots extends AbstractJavaScriptComponent {
045
046        /**
047         * 
048         */
049        private static final long serialVersionUID = 3689118911476002692L;
050
051        private final ArrayList<SvgDots.Dot> dots = new ArrayList<SvgDots.Dot>();
052
053        /**
054         * a SvgDots EventManager
055         */
056        private final EventManager<String, SvgDots, SvgDots.Dot> dotEventManager = new EventManager<String, SvgDots, SvgDots.Dot>();
057
058        /**
059         * SvgDotsListener interface for listening to SvgDot Events fired by an {@link SvgDots.Dot} component.
060         * 
061         * @author Sami Holck
062         * @author Juha Mäkilä
063         * @version v1.0
064         * @since 11.10.2013
065         */
066        public interface SvgDotListener extends EventListener<String, SvgDots, SvgDots.Dot> {
067                /**
068                 * click event (Fired whenever a dot has been clicked)
069                 */
070                public static String CLICK_EVENT = "click";
071
072                /**
073                 * Called in response to a {@link EventManager#callListeners() } call if the present listener is registered to listen for the given event.
074                 * 
075                 * @param eventName Name of the event.
076                 * @param source the source SVG canvas of the SvgDot Event.
077                 * @param dot the dot that triggered the event.
078                 */
079                @Override
080                public void on(String eventName, SvgDots source, SvgDots.Dot dot);
081        }
082
083        /**
084         */
085        public SvgDots() {
086                this(0);
087                this.initListeners();
088        }
089
090        /**
091         * Creates amount of <code>numDots</code> Silver grey colored dots.
092         * 
093         * @.pre numDots >= 0
094         * @param numDots number of the dots.
095         */
096        public SvgDots(int numDots) {
097                this(numDots, "Silver");
098        }
099
100        /**
101         * Creates an amount of <code>numDots</code> SVG dots of color (<code>color</code>).
102         * 
103         * @.pre numDots >= 0
104         * @param numDots number of the dots.
105         * @param color color of the dots.
106         */
107        public SvgDots(int numDots, String color) {
108                ArrayList<SvgDots.Dot> dotsData = new ArrayList<SvgDots.Dot>();
109                for (int i = 0; i < numDots; i++) {
110                        dotsData.add(new SvgDots.Dot(i, color));
111                }
112                this.draw(dotsData);
113                this.initListeners();
114        }
115
116        /**
117         * Draws an amount of <code>dots.size()</code> SVG dots to the SVG canvas.
118         * 
119         * @param dots the list of the SVG dots.
120         */
121        public SvgDots(final Collection<SvgDots.Dot> dots) {
122                this.draw(dots);
123                this.initListeners();
124        }
125
126        private void initListeners() {
127                this.setId("svg_" + this.hashCode());
128                this.addFunction("delegateDotEvent", new JavaScriptFunction() {
129                        /**
130                         * version control in a Serializable class
131                         */
132                        private static final long serialVersionUID = -4245967454493315292L;
133
134                        @Override
135                        public void call(JSONArray arguments) throws JSONException {
136                                if (arguments.length() > 1) {
137                                        // System.out.println("dotEvent");
138                                        // System.out.println("arguments" + arguments);
139                                        dotEventManager.callListeners(arguments.getString(0), SvgDots.this, dots.get(arguments.getInt(1)));
140                                }
141                        }
142                });
143        }
144
145        /**
146         * Draws the given SVG dots to the SVG canvas.
147         * 
148         * <p>
149         * <strong class="red">NOTE:</strong> removes all previously defined SVG dots.
150         * </p>
151         * 
152         * @param dotList
153         */
154        public void draw(Collection<SvgDots.Dot> dotList) {
155                this.dots.clear();
156                this.dots.addAll(dotList);
157                this.callFunction("draw", this.dots);
158                //getState().dots = this.dots;
159                //getState().task = SvgDotsState.REPAINT_ALL;
160                // getState().dotData = a;
161        }
162
163        /**
164         * Replaces the dot at the given index.
165         * 
166         * @param index 
167         * @param dot 
168         * @throws IndexOutOfBoundsException - if the index is out of range (index < 0 || index >= size())
169         */
170        public void replace(int index, SvgDots.Dot dot) {
171                this.dots.remove(index);
172                this.dots.add(index, dot);
173                this.callFunction("repaintDot", index, dot);
174        }
175
176        /**
177         * Returns the dot of the given index.
178         * 
179         * @param index the index of the dot.
180         * @return the dot of the given index.
181         * @throws IndexOutOfBoundsException - if the index is out of range (index < 0 || index >= {@link #size()})
182         */
183        public final Dot getDot(int index) {
184                return this.dots.get(index);
185        }
186
187        /**
188         * Returns the number of SVG dots drawn into this {@link SvgDots}.
189         * 
190         * @.post RESULT >= 0
191         * @return the number of dots in this {@link SvgDots} Component
192         */
193        public int size() {
194                return this.dots.size();
195        }
196
197        /**
198         * Adds a new {@link SvgDotListener} listener into the {@link SvgDots} component.
199         * 
200         * @param eventName The Video.js event the listener will listen to.
201         * @param listener The event listener object itself.
202         */
203        public void addDotsListener(String eventName, SvgDotListener listener) {
204                this.dotEventManager.addListener(eventName, listener);
205        }
206
207        /**
208         * Remove listener from a specific {@link SvgDotListener} event.
209         * 
210         * @param eventName the name of the {@link SvgDotListener} event.
211         * @param listener {@link SvgDotListener} Event listener to remove.
212         */
213        public void removeListener(String eventName, SvgDotListener listener) {
214                this.dotEventManager.removeListener(eventName, listener);
215        }
216
217        /**
218         * Remove listener from all {@link SvgDotListener} events it is registered by. 
219         * Convenient way of cleaning up an listener object being destroyed.
220         * 
221         * @param listener {@link SvgDotListener} Event listener to remove.
222         */
223        public void removeListener(SvgDotListener listener) {
224                this.dotEventManager.removeListener(listener);
225        }
226
227        @Override
228        protected SvgDotsState getState() {
229                return (SvgDotsState) super.getState();
230        }
231
232        /**
233         * Dot contains all the required information about a single SVG dot in {@link SvgDots} component.
234         * 
235         * <p>
236         * <strong class="red">Note:</strong> Dot is a JavaBeans Component.
237         * </p>
238         * 
239         * @author Sami Holck
240         * @author Juha Mäkilä
241         * @version v1.01
242         * @since 22.10.2013
243         * 
244         */
245        public static class Dot implements Serializable {
246
247                /**
248                 * 
249                 */
250                private static final long serialVersionUID = 5743971405439331162L;
251
252                /**
253                 * Property <code>title</code>: the title attribute of the SVG dot
254                 */
255                private String title = "";
256
257                /**
258                 * Property <code>color</code>: the color of the SVG dot
259                 */
260                private String color = "Silver";
261
262                private boolean clickable = true;
263
264
265                /**
266                 * 
267                 */
268                public Dot() {
269                        this("Silver");
270                }
271
272                /**
273                 * 
274                 * @param index
275                 * @param color the color of the SVG dot.
276                 */
277                public Dot(String color) {
278                        this(color, "");
279                }
280
281                /**
282                 * 
283                 * @param color the color of the SVG dot.
284                 * @param title the title attribute of the SVG dot.
285                 */
286                public Dot(String color, String title) {
287                        this.title = title;
288                        this.color = color;
289                }
290
291                /**
292                 * Returns the title attribute of the SVG dot.
293                 * 
294                 * @return the title attribute of the SVG dot.
295                 */
296                public String getTitle() {
297                        return this.title;
298                }
299
300                /**
301                 * Sets the title attribute of the SVG dot.
302                 * 
303                 * @param title the title to set
304                 */
305                public void setTitle(String title) {
306                        this.title = title;
307                }
308
309                /**
310                 * Returns the color of the SVG dot.
311                 * 
312                 * @return the color of the SVG dot.
313                 */
314                public String getColor() {
315                        return this.color;
316                }
317
318                /**
319                 * Sets the color of the SVG dot.
320                 * 
321                 * @param color the color to set
322                 */
323                public final void setColor(String color) {
324                        this.color = color;
325                }
326
327                /**
328                 * Checks if the dot is clickable or not.
329                 * 
330                 * @return true if the dot is clickable and false otherwise.
331                 */
332                public boolean isClickable() {
333                        return clickable;
334                }
335
336                /**
337                 * Changes the clickable state of the dot.
338                 * 
339                 * @param clickable a boolean value specifying whether the dot is clickable or not
340                 */
341                public final void setClickable(boolean clickable) {
342                        this.clickable = clickable;
343                }
344
345                @Override
346                public boolean equals(Object o) {
347                        if (this == o)
348                                return true;
349                        if (!(o instanceof Dot))
350                                return false;
351                        Dot that = (Dot) o;
352                        return new EqualsBuilder()
353                                .append(this.title, that.title)
354                                .append(this.color, that.color).isEquals();
355                }
356                
357                @Override
358                public int hashCode() {
359                return new HashCodeBuilder(17, 31). // two randomly chosen prime numbers
360                    // if deriving: appendSuper(super.hashCode()).
361                    append(this.title).
362                    append(this.color).
363                    toHashCode();
364            }
365
366
367        }
