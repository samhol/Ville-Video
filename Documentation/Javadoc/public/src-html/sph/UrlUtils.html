022public class UrlUtils {
023        
024        /**
025         * Tests if the given HTTP URL is working.
026         * 
027         * @param url the tested HTTP URL
028         * @return true if the given URL exists and false otherwise.
029         */
030        public static boolean exists(String url) {
031                try {
032                        HttpURLConnection.setFollowRedirects(false);
033                        // note : you may also need
034                        // HttpURLConnection.setInstanceFollowRedirects(false);
035                        HttpURLConnection con = (HttpURLConnection) new URL(url).openConnection();
036                        con.setRequestMethod("HEAD");
037                        return (con.getResponseCode() == HttpURLConnection.HTTP_OK);
038                } catch (Exception e) {
039                        // System.out.println("HTTP url: '" + url + "' is not valid");
040                        // e.printStackTrace();
041                        return false;
042                }
043        }
044}
