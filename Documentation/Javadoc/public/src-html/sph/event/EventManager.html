017package sph.event;
018
019/**
020 * Generic event manager interface. Events are of any type. Only the negotiating
021 * classes needs to know about the events existence. It is transparent also to
022 * the EventManager who only delegates the events.
023 * 
024 * <p class="red"><strong>Note:</strong> great care must be exercised if mutable
025 * objects are used as events. The behavior of a map is not specified if the value
026 * of an object is changed in a manner that affects equals comparisons while the
027 * object is a key in the map.</p>
028 * <p>
029 * A potential problem with string based events are that there might be name
030 * conflicts. In a large system, one might consider prefixing the event names to
031 * avoid conflicts. If this seems to become a problem, the system is probably
032 * bad designed anyway; The number of different event types should be kept low.
033 * <p>
034 * The listener class must implement the {@link sph.event.EventListener} interface
035 * which implies the {@link sph.event.EventListener#on(E event, S source, D data)}
036 * method. The listener class registers itself in the
037 * EventManager
038 * by:
039 * 
040 * <pre>
041 * EventManager e = new EventManager();
042 * e.addListener(&quot;EventName&quot;, source);
043 * </pre>
044 * 
045 * The <em>source</em> class of the event will call:
046 * 
047 * <pre>
048 * e.callListeners(&quot;EventName&quot;, object, data);
049 * </pre>
050 * and the EventManager will then call the {@link sph.event.EventListener#on()} method
051 * of every listener.
052 * <p>
053 * The definition of <em>event</em>, <em>object</em> and <em>data</em> is purely up to the
054 * involved classes. Typically <em>object</em> will be the source of the event
055 * (the <em>created</em> object for a "Create" event, the <em>deleted</em>
056 * object for a "Delete" event and so forth.) The additional <em>data</em> object
057 * is for convenience only and will often be null.
058 *
059 * @author  <a href="mailto:sami.holck@gmail.com">Sami Holck</a>
060 * @version 0.1
061 * @since   07.11.2013
062 * @param   <E> the parameterized event type.
063 * @param   <S> the parameterized type of the event source.
064 * @param   <D> the parameterized type of the event data.
065 * @param   <L> the parameterized type of the event listener.
066 */
067public interface EventManager<E, S, D, L extends EventListener<E, S, D>>  {
068
069        /**
070         * Add a listener to the manager.
071         * 
072         * @param event The event the listener will listen to.
073         * @param eventListener The event listener object itself.
074         */
075        public void addListener(E event, L eventListener);
076
077        /**
078         * Remove listener from specific event.
079         * 
080         * @param event Event object to remove listener from.
081         * @param eventListener Listener to remove.
082         */
083        public void removeListener(E event, L eventListener);
084
085        /**
086         * Remove listener from all events it is registered by. Convenient way of
087         * cleaning up an listener object being destroyed.
088         * 
089         * @param eventListener Event listener to remove.
090         */
091        public void removeListener(L eventListener);
092
093        /**
094         * Clears all EventListeners from the manager.
095         */
096        public void clear();
097
098        /**
099         * Calls listeners. The definition of <em>event</em>, <em>source</em> and <em>data</em> is
100         * purely up to the communicating classes.
101         * 
102         * @param event the event object.
103         * @param source Source of the event object (or null).
104         * @param data Additional data of the event (or null).
105         */
106        public void callListeners(E event, S source, D data);
107
108        /**
109         * Calls listeners. Convenience front-end where the additional data parameter is null.
110         * The definition of <em>event</em> and <em>source</em> is purely up to the
111         * communicating classes.
112         * 
113         * @param event the event object.
114         * @param source Source of the event object (or null).
115         */
116        public void callListeners(E event, S source);
117}
