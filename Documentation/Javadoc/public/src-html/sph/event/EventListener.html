019package sph.event;
020
021import java.io.Serializable;
022
023/**
024 * Interface shared by all <em>listeners</em> handled in an {@link BasicEventManager} object.
025 * 
026 * @author <a href="mailto:info@geosoft.no">GeoSoft</a>
027 * @param <E> the parameterized event type.
028 * @param <S> the parameterized type of the event source.
029 * @param <D> the parameterized type of the event data.
030 */
031public interface EventListener<E, S, D> extends Serializable {
032
033        /**
034         * Called in response to a {@link BasicEventManager#callListeners(E eventName, S source, D data)
035         *  EventManager.callListeners(E, S, T) } call if the
036         * present listener is registered to listen for the given event.
037         * 
038         * @param event  The event object fired.
039         * @param source The source the event (as defined by caller).
040         * @param data   Additional data (as defined by caller).
041         */
042        public void on (E event, S source, D data);
043}
