017package sph.event;
018
019import java.io.Serializable;
020import java.util.ArrayList;
021import java.util.Collections;
022import java.util.HashMap;
023import java.util.HashSet;
024import java.util.Iterator;
025import java.util.Map;
026import java.util.Set;
027
028/**
029 * Generic event manager class. Events are of any type. Only the negotiating
030 * classes needs to know about the events existence. It is transparent also to
031 * the EventManager who only delegates the events.
032 * 
033 * <p class="red"><strong>Note:</strong> great care must be exercised if mutable
034 * objects are used as events. The behavior of a map is not specified if the value
035 * of an object is changed in a manner that affects equals comparisons while the
036 * object is a key in the map.</p>
037 * <p>
038 * A potential problem with string based events are that there might be name
039 * conflicts. In a large system, one might consider prefixing the event names to
040 * avoid conflicts. If this seems to become a problem, the system is probably
041 * bad designed anyway; The number of different event types should be kept low.
042 * <p>
043 * The listener class must implement the {@link EventListener} interface
044 * which implies the {@link sph.event.EventListener#on(E event, S source, D data)}
045 * method. The listener class registers itself in the
046 * EventManager
047 * by:
048 * 
049 * <pre>
050 * EventManager e = new EventManager();
051 * e.addListener(&quot;EventName&quot;, source);
052 * </pre>
053 * 
054 * The <em>source</em> class of the event will call:
055 * 
056 * <pre>
057 * e.callListeners(&quot;EventName&quot;, object, data);
058 * </pre>
059 * and the EventManager will then call the {@link sph.event.EventListener#on()} method
060 * of every listener.
061 * <p>
062 * The definition of <em>event</em>, <em>object</em> and <em>data</em> is purely up to the
063 * involved classes. Typically <em>object</em> will be the source of the event
064 * (the <em>created</em> object for a "Create" event, the <em>deleted</em>
065 * object for a "Delete" event and so forth.) The additional <em>data</em> object
066 * is for convenience only and will often be null.
067 * 
068 * @author <a href="mailto:sami.holck@gmail.com">Sami Holck</a>
069 * @param <E> the parameterized event type.
070 * @param <S> the parameterized type of the event source.
071 * @param <D> the parameterized type of the event data.
072 * @param <L> the parameterized type of the event listener.
073 */
074public class BasicEventManager<E, S, D, L extends EventListener<E, S, D>> implements EventManager<E, S, D, L>, Serializable {
075
076        private static final long serialVersionUID = -2199047086769625113L;
077        private final Map<E, Set<L>> listeners_;
078
079        /**
080         * Create the event manager instance.
081         */
082        public BasicEventManager() {
083                listeners_ = Collections.synchronizedMap(new HashMap<E, Set<L>>());
084        }
085
086        /**
087         * Add a listener.
088         * 
089         * @param event The event the listener will listen to.
090         * @param eventListener The event listener object itself.
091         */
092        @Override
093        public void addListener(E event, L eventListener) {
094                // Check if this is a new event name
095                Set<L> eventListeners = listeners_.get(event);
096                if (eventListeners == null) {
097                        eventListeners = new HashSet<L>();
098                        listeners_.put(event, eventListeners);
099                }
100                // Add the listener
101                eventListeners.add(eventListener);
102        }
103
104        /**
105         * Remove listener from specific event.
106         * 
107         * @param event Event object to remove listener from.
108         * @param eventListener Listener to remove.
109         */
110        @Override
111        public void removeListener(E event, L eventListener) {
112                if (event == null) {
113                        removeListener(eventListener);
114                        return;
115                }
116
117                // Find the listeners for the specified event
118                Set<L> eventListeners = listeners_.get(event);
119                if (eventListeners == null) {
120                        return;
121                }
122
123                // Remove the listener
124                eventListeners.remove(eventListener);
125                // Remove the event as such if this was the last listener for this event
126                if (eventListeners.size() == 0) {
127                        listeners_.remove(event);
128                }
129        }
130
131        /**
132         * Remove listener from all events it is registered by. Convenient way of
133         * cleaning up an listener object being destroyed.
134         * 
135         * @param eventListener Event listener to remove.
136         */
137        @Override
138        public void removeListener(L eventListener) {
139                // Loop over all registered events and remove the specified listener
140                // Loop over a copy incase the removeListener() call wants to
141                // remove the entire event from the hash map.
142                ArrayList<E> listeners = new ArrayList<E>(listeners_.keySet());
143                for (Iterator<E> i = listeners.iterator(); i.hasNext();) {
144                        E eventName = i.next();
145                        removeListener(eventName, eventListener);
146                }
147        }
148
149        /**
150         * Clears all EventListeners from the manager.
151         */
152        @Override
153        public void clear() {
154                listeners_.clear();
155        }
156
157        /**
158         * Calls listeners. The definition of <em>event</em>, <em>source</em> and <em>data</em> is
159         * purely up to the communicating classes.
160         * 
161         * @param event the event object.
162         * @param source Source of the event object (or null).
163         * @param data Additional data of the event (or null).
164         */
165        @Override
166        public void callListeners(E event, S source, D data) {
167                // Find all listeners of this event
168                Set<L> eventListeners = listeners_.get(event);
169                if (eventListeners == null) {
170                        return;
171                }
172
173                // Loop over a copy of the set in case it is altered by listener
174                Set<L> copy = new HashSet<L>(eventListeners);
175                for (Iterator<L> i = copy.iterator(); i.hasNext();) {
176                        L eventListener = i.next();
177                        if (eventListener == null) {
178                                i.remove();
179                        } else {
180                                eventListener.on(event, source, data);
181                        }
182                }
183        }
184
185        /**
186         * Calls listeners. Convenience front-end where the additional data parameter is null.
187         * The definition of <em>event</em> and <em>source</em> is purely up to the
188         * communicating classes.
189         * 
190         * @param event the event object.
191         * @param source Source of the event object (or null).
192         */
193        @Override
194        public void callListeners(E event, S source) {
195                callListeners(event, source, null);
196        }
197}
