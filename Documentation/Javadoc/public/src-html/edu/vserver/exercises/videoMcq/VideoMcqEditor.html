037public class VideoMcqEditor extends HorizontalLayout implements Editor<VideoMcqExerciseData> {
038
039        private static final long serialVersionUID = -4600841604409240872L;
040        private EditorHelper<VideoMcqExerciseData> exerInfoEditor;
041        private VideoMcqExerciseData oldExerData;
042
043        private final Panel qEditPanel = new Panel("Saved Questions:");
044        private VerticalLayout questionEditButtons;
045
046        private Panel exerPanel;
047        private final VideojsFrame videojsFrame = new VideojsFrame();
048        private Videojs videojs;
049        private final Panel videoPanel = new Panel();
050        private QuestionLibrary questionLibrary;
051
052        private CheckBox questionSkippingCheckBox;
053        private CheckBox instRespCheckBox;
054        private QuestionForm questionForm;
055
056
057        @Override
058        public Layout getView() {
059                return this;
060        }
061
062        /**
063         * 
064         * @see edu.vserver.exercises.model.Editor#initialize(edu.vserver.standardutils.Localizer,
065         *      edu.vserver.exercises.model.ExerciseData,
066         *              edu.vserver.exercises.model.EditorHelper)
067         */
068        @Override
069        public void initialize( Localizer localizer, VideoMcqExerciseData ExerData, EditorHelper<VideoMcqExerciseData> genExerInfoEditor ) {
070
071                //this.localizer = localizer;
072                this.exerInfoEditor = genExerInfoEditor;
073                this.oldExerData = ( ExerData == null ? new VideoMcqExerciseData() : ExerData );
074                this.questionLibrary = this.oldExerData.getQuestionLibrary();
075                doLayout( this.oldExerData );
076        }
077
078        private VideoMcqExerciseData getCurrentExercise() {
079                System.out.println(this.questionLibrary);
080                VideoMcqExerciseData exerData = new VideoMcqExerciseData();
081                exerData.setExerName(this.exerInfoEditor.getExerciseInfo().getName());
082                exerData.setVideoURL(videojs.getVideoSrc());
083                exerData.setMimeType(videojs.getMimeType());
084                exerData.setQuestionLibrary(questionLibrary);
085                exerData.setInstantResponse(this.instRespCheckBox.getValue());
086                exerData.setSliderEnabled(this.questionSkippingCheckBox.getValue());
087                return exerData;
088        }
089
090        private void doLayout( VideoMcqExerciseData oldExerData ) {
091
092                this.setSpacing( true );
093
094                videojsFrame.setVideoSource( oldExerData.getVideoURL(), oldExerData.getMimeType() );
095                videojs = videojsFrame.getVideojsPlayer();
096                videoPanel.setContent( videojsFrame );
097                VerticalLayout controlsLayout = new VerticalLayout();
098                controlsLayout.setWidth(250, Sizeable.Unit.PIXELS);
099                controlsLayout.addComponent( exerInfoEditor.getInfoEditorView() );
100                controlsLayout.addComponent( exerInfoEditor.getControlbar( new EditedExerciseGiver< VideoMcqExerciseData >() {
101
102                        @Override
103                        public VideoMcqExerciseData getCurrExerData( boolean forSaving ) {
104                                return getCurrentExercise();
105                        }
106                }));
107
108                this.addComponent( controlsLayout );
109                exerPanel = new Panel();
110                VerticalLayout exerLayout = new VerticalLayout();
111                exerPanel.setContent( exerLayout );
112                exerLayout.setSpacing( true );
113
114                this.questionSkippingCheckBox = new CheckBox("Allow skipping questions");
115                this.questionSkippingCheckBox.setValue( oldExerData.isSliderEnabled() );
116                this.instRespCheckBox = new CheckBox("Give instant response");
117                this.instRespCheckBox.setValue(oldExerData.isInstantResponse());
118
119                HorizontalLayout globalVariableLayout = new HorizontalLayout();
120                globalVariableLayout.setSpacing(true);
121                globalVariableLayout.addComponents(new Label("<strong>Global settings: </strong>", ContentMode.HTML), this.questionSkippingCheckBox, this.instRespCheckBox);
122                this.questionForm = new QuestionForm(videojs);
123                exerLayout.addComponents(videoPanel, globalVariableLayout, this.questionForm );
124
125                this.setupSavedQuestionsListView();
126                this.addComponents(exerLayout, qEditPanel);
127                this.repaintSavedQuestionsListView();
128                this.questionForm.addQuestionFormListener(QuestionForm.QuestionFormListener.QUESTION_EDITED_EVENT,
129                                new QuestionForm.QuestionFormListener() {
130
131                        private static final long serialVersionUID = 8674189273630233196L;
132
133                        /**
134                         * {@link QuestionForm} component's
135                         * 
136                         * @param eventName name of the {@link QuestionWindow.VjsListener VideojsListener} event.
137                         * @param source the source object of the {@link QuestionFormListener VideojsListener} event.
138                         * @param question the question currently handled.
139                         */
140                        @Override
141                        public void on(String eventName, QuestionForm source, Question question) {
142                                Question old = source.getOriginalQuestion();
143                                questionLibrary.replace( old, question );
144                                repaintSavedQuestionsListView();
145                        }
146                });
147
148                this.questionForm.addQuestionFormListener(QuestionForm.QuestionFormListener.QUESTION_CREATED_EVENT,
149                                new QuestionForm.QuestionFormListener() {
150
151                        private static final long serialVersionUID = 8674189273630233196L;
152
153                        /**
154                         * {@link QuestionWindow} Component has triggered an event.
155                         * 
156                         * @param eventName name of the {@link QuestionWindow.VjsListener VideojsListener} event.
157                         * @param source the source object of the {@link QuestionFormListener VideojsListener} event.
158                         * @param question the question currently handled.
159                         */
160                        @Override
161                        public void on(String eventName, QuestionForm source, Question question) {
162                                questionLibrary.add( question );
163                                repaintSavedQuestionsListView();
164                        }
165                });
166        }
167
168        /**
169         * Creates an edit button with click action (Click action: sets the QuestionForm to edit the question)
170         * 
171         * @param  question the question.
172         * @return created button.
173         */
174        private Button createEditButton(Question question) {
175                Button btn = ComponentFactory.createSmallBlueButton("Edit", true, SPH_Theme.EDIT_ICON_16PX);
176                btn.setDescription("<h4 class=\"tooltip\">EDIT QUESTION</h4>" +
177                                "<p class=\"tooltip\">" + question.getQuestion() + "</p>");
178                btn.setWidth(60, Unit.PIXELS);
179                btn.setData(question);
180                btn.addClickListener(new Button.ClickListener() {
181
182                        private static final long serialVersionUID = -2352982251747900119L;
183
184                        @Override
185                        public void buttonClick(ClickEvent event) {
186                                Question q = (Question) event.getButton().getData();
187                                questionForm.editQuestion( q );
188                        }
189                });
190                return btn;
191        }
192
193        /**
194         * Creates a delete button with click action (Click action: removes the row and sets the deletion dialog)
195         * 
196         * @param  question the question.
197         * @return created button.
198         */
199        private Button createRemoveButton(Question question) {
200                Button btn = ComponentFactory.createSmallBlueButton("Del", true, SPH_Theme.DELETE_ICON_16PX);
201                btn.setDescription("<h4 class=\"redText tooltip\">DELETE QUESTION?</h4>" +
202                                "<p class=\"tooltip\">" + question.getQuestion() + "</p>");
203                btn.setWidth(60, Unit.PIXELS);
204                btn.setData(question);
205                btn.addClickListener(new Button.ClickListener() {
206
207                        private static final long serialVersionUID = -6016940629549296901L;
208
209                        @Override
210                        public void buttonClick(ClickEvent event) {
211                                Button b = event.getButton();
212                                QuestionRemovalDialog dialog = new QuestionRemovalDialog((Question) b.getData(), new QuestionDeleter(b));
213                                dialog.show();
214                        }
215                });
216                return btn;
217        }
218
219        /**
220         * The action performed when the delete button gets clicked
221         *
222         * @author  <a href="mailto:sami.holck@gmail.com">Sami Holck</a>
223         * @author  Juha Mäkilä
224         * @version 0.1
225         * @since   8.11.2013
226         */
227        private class QuestionDeleter implements ConfirmDialog.Callback {
228                private final Button delBtn;
229                /**
230                 * 
231                 * @param b the target delete button
232                 */
233                public QuestionDeleter(Button b) {
234                        delBtn = b;
235                }
236
237                @Override
238                public void onDialogResult(boolean resultIsYes) {
239                        if (resultIsYes) {
240                                Question q = (Question) delBtn.getData();
241                                questionLibrary.remove(q);
242                                HorizontalLayout row = (HorizontalLayout) delBtn.getParent();
243                                questionEditButtons.removeComponent(row);
244                                repaintSavedQuestionsListHeading(questionLibrary.size());
245                                questionForm.clearForm();
246                        }
247                }
248        }
249
250        /**
251         * Set the saved questions list view component ready for use.
252         */
253        private void setupSavedQuestionsListView() {
254                this.qEditPanel.setEnabled(false);
255                this.questionEditButtons = new VerticalLayout();
256                this.questionEditButtons.setSpacing(true);
257                this.questionEditButtons.setMargin(true);
258                Panel qEditScrollPanel = new Panel();
259                qEditScrollPanel.setHeight(490, Unit.PIXELS);
260                qEditScrollPanel.setWidth(220, Unit.PIXELS);
261                qEditScrollPanel.setContent(this.questionEditButtons);
262                this.qEditPanel.setContent(qEditScrollPanel);
263        }
264
265        /**
266         * Repaints the contents of the saved questions list view component and
267         * enables it to the editor view if it contains any Questions.
268         */
269        private void repaintSavedQuestionsListView() {
270                int no = this.questionLibrary.size();
271                if (no > 0) {
272                        this.repaintSavedQuestionsListHeading(no);
273                        this.questionEditButtons.removeAllComponents();
274                        for (Question q : this.questionLibrary) {
275                                this.questionEditButtons.addComponent(createSavedQuestionRow(q));
276                        }
277                        this.qEditPanel.setEnabled(true);
278                } else {
279                        this.qEditPanel.setEnabled(false);
280                }
281        }
282
283
284        /**
285         * Repaints the heading text of the saved questions list view component.
286         * 
287         * @param qCount number of the saved questions.
288         */
289        private void repaintSavedQuestionsListHeading(int qCount) {
290                String text = "Saved Questions: <em style=\"font-size: 60%;\">" + qCount;
291                text += (qCount > 1)? " units</em>" : " unit</em>";
292                this.qEditPanel.setCaption(text);
293        }
294
295        /**
296         * Creates a row ({@link HorizontalLayout}) for new saved {@link Question}.
297         * 
298         * <p>the row contains edit and delete buttons for the saved {@link Question}.</p>
299         * 
300         * @param question target question
301         * @return the row created
302         */
303        private HorizontalLayout createSavedQuestionRow(Question question) {
304                HorizontalLayout hl = new HorizontalLayout();
305                hl.setSpacing( true );
306                Label timeLabel = new Label( Time.convertToTimeString( question.getTime() ) );
307                Button editButton = createEditButton( question );
308                Button removeButton = createRemoveButton( question );
309                hl.addComponents( timeLabel, editButton, removeButton );
310                return hl;
311        }
