001package edu.vserver.exercises.videoMcq;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import com.vaadin.ui.Alignment;
007import com.vaadin.ui.Component;
008import com.vaadin.ui.Label;
009import com.vaadin.ui.VerticalLayout;
010
011import edu.vserver.exercises.helpers.StatSubmInfoFilterEditor;
012import edu.vserver.exercises.helpers.StatSubmInfoFilterEditor.FilterEditorFactory;
013import edu.vserver.exercises.helpers.StatSubmInfoFilterTable;
014import edu.vserver.exercises.helpers.StatSubmInfoFilterTable.ShowSubmissionColGenerator;
015import edu.vserver.exercises.helpers.StatSubmInfoFilterTable.SubmInfoColumnGenerator;
016import edu.vserver.exercises.model.ExerciseException;
017import edu.vserver.exercises.model.StatisticalSubmissionInfo;
018import edu.vserver.exercises.model.SubmissionStatisticsGiver;
019import edu.vserver.standardutils.Localizer;
020import edu.vserver.standardutils.TempFilesManager;
021
022/**
023 * Statistics giver.
024 *
025 * @author  <a href="mailto:sami.holck@gmail.com">Sami Holck</a>
026 * @author  Juha Mäkilä
027 * @version 1.1
028 * @since   16.10.2013
029 */
030public class VideoMcqSubmissionStatisticsGiver extends VerticalLayout implements SubmissionStatisticsGiver<VideoMcqExerciseData, VideoMcqSubmissionInfo> {
031
032        private static final long serialVersionUID = -1410253605264134011L;
033
034        private List<StatisticalSubmissionInfo<VideoMcqSubmissionInfo>> data;
035        private VideoMcqExerciseData exer;
036
037        private StatSubmInfoFilterTable<VideoMcqSubmissionInfo> genTable;
038        private StatSubmInfoFilterEditor<VideoMcqSubmissionInfo> genTableEditor;
039
040        private Localizer localizer;
041
042        @Override
043        public void initialize(
044                        VideoMcqExerciseData exercise,
045                        List<StatisticalSubmissionInfo<VideoMcqSubmissionInfo>> dataObjects,
046                        Localizer localizer, TempFilesManager tempManager)
047                                        throws ExerciseException {
048                exer = exercise;
049                data = dataObjects;
050                this.localizer = localizer;
051                initAllSubmissionsTable();
052
053                doLayout();
054        }
055
056
057        private void initAllSubmissionsTable() {
058
059                List<SubmInfoColumnGenerator<?, VideoMcqSubmissionInfo>> extraCols = new ArrayList<SubmInfoColumnGenerator<?, VideoMcqSubmissionInfo>>();
060
061                //extraCols.add(new AnswerColGen());
062                // using null for temp-files-manager is safe here as it is known
063                // that TemplateSubmissionViewer does not need TempFilesManager;
064                // though it would not hurt to pass the real TempFilesManager anyway
065                extraCols.add(new ShowSubmissionColGenerator<VideoMcqExerciseData, VideoMcqSubmissionInfo>(
066                                VideoMcqDescriptor.INSTANCE, exer, null));
067
068                genTable = new StatSubmInfoFilterTable<VideoMcqSubmissionInfo>(
069                                localizer, data, extraCols);
070
071                List<FilterEditorFactory<VideoMcqSubmissionInfo>> extras = new ArrayList<FilterEditorFactory<VideoMcqSubmissionInfo>>();
072
073                genTableEditor = new StatSubmInfoFilterEditor<VideoMcqSubmissionInfo>(genTable, localizer, extras);
074
075        }
076
077        private void doLayout() {
078                this.setWidth("100%");
079
080                VerticalLayout centeredLayout = new VerticalLayout();
081                centeredLayout.setWidth("800px");
082
083                Label questionLabel = new Label("Video exercise: " + exer.getExerName());
084                questionLabel.addStyleName("template-exercise-title");
085                centeredLayout.addComponent(questionLabel);
086                centeredLayout.addComponent(genTableEditor.getView());
087                centeredLayout.addComponent(genTable.getStatInfoTableView());
088
089                this.addComponent(centeredLayout);
090                this.setComponentAlignment(centeredLayout, Alignment.TOP_CENTER);
091
092        }
093
094        @Override
095        public Component getView() {
096                return this;
097        }
098
099        @Override
100        public String exportStatisticsDataAsText() {
101                // TODO Auto-generated method stub
102                return "exportStatisticsDataAsText";
103        }
104
105}
