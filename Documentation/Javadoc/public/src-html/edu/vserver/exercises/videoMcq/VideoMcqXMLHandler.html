026/**
027 * Class writes and reads the videoMcq XML data.
028 * 
029 * @author  Juha Mäkilä

034public final class VideoMcqXMLHandler implements PersistenceHandler<VideoMcqExerciseData, VideoMcqSubmissionInfo> {
035
036        private static final long serialVersionUID = -3150033834959652936L;
037        /**
038         * Gives a new instance of VideoMcqXMLHandler.
039         */
040        public static final VideoMcqXMLHandler INSTANCE = new VideoMcqXMLHandler();
041
042        private static final String VIDEO_URL = "videoURL";
043        private static final String SLIDER_ENABLED = "sliderEnabled";
044
045        private VideoMcqXMLHandler() {
046
047        }
048
049        @Override
050        public VideoMcqExerciseData loadExerData(byte[] dataPres, TempFilesManager tempManager) throws ExerciseException {
051                VideoMcqExerciseData exerData = new VideoMcqExerciseData();
052                try {
053                        Document doc = XMLHelper.parseFromBytes(dataPres);
054                        doc.getDocumentElement().normalize();
055                        exerData.setVideoURL(doc.getDocumentElement().getAttribute(VIDEO_URL));
056                        exerData.setMimeType(doc.getDocumentElement().getAttribute("mimetype"));
057                        exerData.setExerName(doc.getDocumentElement().getAttribute("exerName"));
058                        exerData.setSliderEnabled(Boolean.parseBoolean(doc.getDocumentElement().getAttribute(SLIDER_ENABLED)));
059                        exerData.setInstantResponse(Boolean.parseBoolean(doc.getDocumentElement().getAttribute("instantResponse")));
060                        QuestionLibrary questionLibrary = new QuestionLibrary();
061                        NodeList questionNodes = doc.getElementsByTagName("timedQuestion");
062                        for (int i = 0; i < questionNodes.getLength(); i++) {
063                                Node questionNode = questionNodes.item(i);
064                                NodeList questionData = questionNode.getChildNodes();
065                                int time = Integer.parseInt(questionData.item(0).getTextContent());
066                                String question = questionData.item(1).getTextContent();
067                                String answer = questionData.item(2).getTextContent();
068                                String answerDescription = questionData.item(3).getTextContent();
069                                ArrayList<String> falseAnswers = new ArrayList<String>();
070                                NodeList falseData = questionData.item(4).getChildNodes();
071                                for (int j = 0; j < falseData.getLength(); j++) {
072                                        falseAnswers.add(falseData.item(j).getTextContent());
073                                }
074                                questionLibrary.add(new Question(time, question, answer, answerDescription, falseAnswers));
075                        }
076                        exerData.setQuestionLibrary(questionLibrary);
077
078                } catch (ParserConfigurationException e) {
079                        throw new ExerciseException(ExerciseException.ErrorType.EXER_LOAD_ERROR, e);
080                } catch (SAXException e) {
081                        throw new ExerciseException(ExerciseException.ErrorType.EXER_LOAD_ERROR, e);
082                } catch (IOException e) {
083                        throw new ExerciseException(ExerciseException.ErrorType.EXER_LOAD_ERROR, e);
084                }
085                return exerData;
086        }
087
088        @Override
089        public byte[] saveExerData(VideoMcqExerciseData toWrite, TempFilesManager tempManager) throws ExerciseException {
090                byte[] res = null;
091                try {
092                        QuestionLibrary questionLibrary = toWrite.getQuestionLibrary();
093                        ArrayList<Question> questions = questionLibrary.getAllQuestions();
094                        Document doc = XMLHelper.createEmptyDocument();
095
096                        Element root = doc.createElement("videoMcq-exercise");
097                        doc.appendChild(root);
098                        root.setAttribute(VIDEO_URL, toWrite.getVideoURL());
099                        root.setAttribute("mimetype", toWrite.getMimeType());
100                        root.setAttribute("exerName", toWrite.getExerName());
101                        root.setAttribute(SLIDER_ENABLED, Boolean.toString(toWrite.isSliderEnabled()));
102                        root.setAttribute("instantResponse", Boolean.toString(toWrite.isInstantResponse()));
103
104                        for (Question q : questions) {
105                                Element timedQuestion = doc.createElement("timedQuestion");
106                                root.appendChild(timedQuestion);
107
108
109                                Element time = doc.createElement("time");
110                                time.setTextContent(Integer.toString(q.getTime()));
111                                timedQuestion.appendChild(time);
112
113                                Element question = doc.createElement("question");
114                                question.setTextContent(q.getQuestion());
115                                timedQuestion.appendChild(question);
116
117                                Element answer = doc.createElement("answer");
118                                answer.setTextContent(q.getCorrectAnswer(0));
119                                timedQuestion.appendChild(answer);
120
121                                Element answerDescription = doc.createElement("AnswerDescription");
122                                answerDescription.setTextContent(q.getAnswerDescription());
123                                timedQuestion.appendChild(answerDescription);
124
125                                Element falseAnswers = doc.createElement("falseAnswers");
126                                timedQuestion.appendChild(falseAnswers);
127
128                                ArrayList<String> falseList = q.getIncorrectAnswers();
129
130                                for (int j = 0; j < q.getIncorrectAnswers().size(); j++) {
131                                        Element falseAnswer = doc.createElement("false" + (j + 1));
132                                        falseAnswer.setTextContent(falseList.get(j));
133                                        falseAnswers.appendChild(falseAnswer);
134                                }
135                        }
136                        res = XMLHelper.xmlToBytes(doc);
137                } catch (ParserConfigurationException e) {
138                        throw new ExerciseException(ExerciseException.ErrorType.EXER_WRITE_ERROR, e);
139                } catch (TransformerConfigurationException e) {
140                        throw new ExerciseException(ExerciseException.ErrorType.EXER_WRITE_ERROR, e);
141                } catch (TransformerException e) {
142                        throw new ExerciseException(ExerciseException.ErrorType.EXER_WRITE_ERROR, e);
143                } catch (TransformerFactoryConfigurationError e) {
144                        throw new ExerciseException(ExerciseException.ErrorType.EXER_WRITE_ERROR, e);
145                }
146                return res;
147        }
148
149        @Override
150        public byte[] saveSubmission(VideoMcqSubmissionInfo subm, TempFilesManager tempManager) throws ExerciseException {
151                try {
152                        ByteArrayOutputStream bytesOut = new ByteArrayOutputStream();
153                        ObjectOutputStream objOut = new ObjectOutputStream(bytesOut);
154                        objOut.writeObject(subm);
155
156                        return bytesOut.toByteArray();
157
158                } catch (IOException e) {
159                        e.printStackTrace();
160                }
161                return null;
162        }
163
164        @Override
165        public VideoMcqSubmissionInfo loadSubmission(byte[] dataPres, boolean forStatGiver, TempFilesManager tempManager)
166                        throws ExerciseException {
167                VideoMcqSubmissionInfo res = null;
168                try {
169                        ObjectInputStream objIn = new ObjectInputStream(new ByteArrayInputStream(dataPres));
170                        res = (VideoMcqSubmissionInfo) objIn.readObject();
171                } catch (IOException e) {
172                        e.printStackTrace();
173                } catch (ClassNotFoundException e) {
174                        e.printStackTrace();
175                }
176                return res;
177        }
