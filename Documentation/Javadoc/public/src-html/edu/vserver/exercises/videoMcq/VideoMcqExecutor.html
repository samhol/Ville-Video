080        /**
081         * Sets the
082         *
083         * @param askingEnabled the askingEnabled to set
084         */
085        private void setAskingEnabled(boolean askingEnabled) {
086                this.askingEnabled = askingEnabled;
087                if (this.askingEnabled) {
088                        this.goToStartBtn.setEnabled(true);
089                        this.vjs.unmask();
090                } else {
091                        this.goToStartBtn.setEnabled(false);
092                        this.vjs.mask();
093                }
094
095        }

097        @Override
098        public void initialize(Localizer localizer, VideoMcqExerciseData exerciseData, VideoMcqSubmissionInfo oldSubm,
099                        TempFilesManager materials, ExecutionSettings fbSettings) throws ExerciseException {
100                this.exerData = exerciseData;
101                this.questionLibrary = exerData.getQuestionLibrary();
102                this.questionWindow.setInformative(this.exerData.isInstantResponse());
103                vjs.pauseAt(this.questionLibrary.getTimeSlots());
104                if ( oldSubm == null ) {
105                        this.submInfo = new VideoMcqSubmissionInfo();
106                }
107                else {
108                        this.submInfo = oldSubm;
109                        //this.submInfo.setPoints(0);
110                }
111                setListeners();
112                doLayout(this.exerData);
113        }
114
115        private void doLayout(VideoMcqExerciseData exerData) {
116                VerticalLayout exerLayout = new VerticalLayout();
117                exerLayout.setDefaultComponentAlignment(Alignment.TOP_CENTER);
118                vjs.setSource(exerData.getVideoURL(), exerData.getMimeType());
119                vjs.mask();
120                vjs.minimalUserControlsEnabled(true);
121                vjs.supportFullscreenMode(false);
122
123                this.videoLayout.addComponents(vjs, this.startButton);
124                this.videoLayout.setWidth(760, Unit.PIXELS);
125                this.videoLayout.setHeight(432, Unit.PIXELS);
126
127                HorizontalLayout buttonLayout = new HorizontalLayout();
128                buttonLayout.setWidth(700, Unit.PIXELS);
129                buttonLayout.setDefaultComponentAlignment(Alignment.MIDDLE_CENTER);
130                this.startButton.addStyleName("ExerStarter");
131                this.startButton.setDescription("Start the excercise");
132
133                buttonLayout.addComponents(this.startButton);
134                this.buildDotLayout();
135                exerLayout.addComponents(videoLayout, this.dotLayout, buttonLayout);
136
137                questionWindow.setQuestionLibrary(questionLibrary);
138                questionDots.draw(questionLibrary);
139                questionDots.setClickable(exerData.isSliderEnabled());
140                questionDots.setInformative(exerData.isInstantResponse());
141
142                this.addComponent(exerLayout);
143
144                JavaScript.getCurrent().execute("VideoMcqExecutor.setStartButton(" + vjs.getId() + ");");
145        }

164        @Override
165        public void askReset() {
166                setAskingEnabled(false);
167                //askingEnabled = true;
168                //submInfo.setPoints(0);
169                //submInfo.increaseTries();
170                questionLibrary.resetGivenAnswers();
171                vjs.clearPauseAtList();
172                vjs.pauseAt(this.questionLibrary.getTimeSlots());
173                vjs.seekTo(0);
174                vjs.mask();
175                execHelper.informResetDefault();
176                questionDots.draw(this.questionLibrary);
177                goToStartBtn.setEnabled(true);
178                fullWindowButton.setEnabled(true);
179                JavaScript.getCurrent().execute("VideoMcqExecutor.showStartButton();");
180        }

187        @Override
188        public void askSubmit(SubmissionType submType) {
189                double score = questionLibrary.getScore();
190                setAskingEnabled(false);
191                execHelper.informOnlySubmit(score, new VideoMcqSubmissionInfo(questionWindow.getQuestionLibrary()), submType, null );
192                vjs.pause();
193                vjs.mask("Exercise completed!");
194
195                vjs.clearPauseAtList();
196
197                goToStartBtn.setEnabled(false);
198                fullWindowButton.setEnabled(false);
199                JavaScript.getCurrent().execute("VideoMcqExecutor.hideStartButton();");
200
201        }

209        private void setListeners() {
210                if (this.exerData.isInstantResponse()) {
211                        this.questionDots.addDotsListener(QuestionDots.QuestionDotListener.CLICK_EVENT, new QuestionDots.QuestionDotListener() {
212
213                                private static final long serialVersionUID = -1592558255318728462L;
214
215                                @Override
216                                public void on(String eventName, SvgDots canvas, Integer dotIndex) {
217                                        if (isAskingEnabled()) {
218                                                SvgDots.Dot dot = canvas.getDot(dotIndex);
219                                                Question q = (Question) dot.getData();
220                                                vjs.seekTo(q.getTime());
221                                        }
222                                }
223                        });
224                }
225                this.startButton.addAttachListener(new AttachListener() {
226
227                        private static final long serialVersionUID = 111802046795345210L;
228
229                        @Override
230                        public void attach(AttachEvent event) {
231                                JavaScript.getCurrent().execute("VideoMcqExecutor.setStartButton(" + vjs.getId() + ");");
232                                //System.out.println("startButton AttachListener()");
233                        }
234
235                });
236                this.startButton.addClickListener(new Button.ClickListener() {
237
238                        private static final long serialVersionUID = 7181459362080977100L;
239
240                        @Override
241                        public void buttonClick(ClickEvent event) {
242                                JavaScript.getCurrent().execute("VideoMcqExecutor.hideStartButton();");
243                                vjs.unmask();
244                                setAskingEnabled(true);
245                                if (!questionWindow.ask(0)) {
246                                        vjs.play();
247                                }
248                        }
249                });
250                fullWindowButton.addClickListener(new Button.ClickListener() {
251
252                        private static final long serialVersionUID = -8152740888003850116L;
253
254                        @Override
255                        public void buttonClick(ClickEvent event) {
256                                vjs.fullWindowEnabled(true);
257                        }
258                });
259                vjs.addVideojsListener(Videojs.VjsListener.PAUSE_EVENT, new Videojs.VjsListener() {
260
261                        private static final long serialVersionUID = -7260155712968515580L;
262
263                        @Override
264                        public void on(String eventName, Videojs source, Double triggerTime) {
265                                if (isAskingEnabled()) {
266                                        //System.out.println("PAUSED at time: " + triggerTime);
267                                        questionWindow.ask(triggerTime.intValue());
268                                }
269                        }
270                });
271                questionWindow.addQuestionEventListener(QuestionWindow.QuestionEventListener.TIMESLOT_FINISHED_EVENT, new QuestionWindow.QuestionEventListener() {
272
273                        private static final long serialVersionUID = 172768819560745819L;
274
275                        @Override
276                        public void on(String eventName, QuestionWindow source, Integer timeSlot) {
277                                if (!questionLibrary.containsUnansweredInTimeSlot(timeSlot)) {
278                                        vjs.removePauseAtTime(timeSlot);
279                                        if (!questionLibrary.containsUnanswerded()) {
280                                                exerciseFinished();
281                                        }
282                                }
283                                questionDots.draw(questionWindow.getQuestionLibrary());
284                                vjs.play();
285                        }
286                });
287                questionWindow.addQuestionEventListener(QuestionWindow.QuestionEventListener.QUESTION_FINISHED_EVENT,
288                                new QuestionWindow.QuestionEventListener() {
289
290                        private static final long serialVersionUID = 172768819560745819L;
291
292                        @Override
293                        public void on(String eventName, QuestionWindow source, Integer timeSlot) {
294                                questionDots.draw(source.getQuestionLibrary());
295                        }
296                });
297                this.registerSubmitListener(new SubmissionListener<VideoMcqSubmissionInfo>() {
298
299                        private static final long serialVersionUID = 3956036740947573677L;
300
301                        @Override
302                        public void submitted(SubmissionResult<VideoMcqSubmissionInfo> submission) {
303                                vjs.clearPauseAtList();
304                                //System.out.println("submitted");
305                        }
306
307                });
308        }

310        /**
311         * Disables exercise controls and gives a tray notification.
312         */
313        private void exerciseFinished() {
314                System.out.println("All questions asked");
315                goToStartBtn.setEnabled(false);
316                fullWindowButton.setEnabled(false);
317                Notification n = new Notification("All " + questionLibrary.size() + " questions are answered!",
318                                Notification.Type.TRAY_NOTIFICATION);
319                n.setIcon(new ThemeResource(SPH_Theme.INFO_ICON_24PX));
320                n.setDelayMsec(500);
321                //n.setPosition(com.vaadin.shared.Position.MIDDLE_CENTER);
322                n.show(Page.getCurrent());
323        }

325        /**
326         * Sets up the dot layout which allows user to see and click question dots.
327         * 
328         * <p>Dot clicking is only allowed if question skipping is allowed in the exercise.</p>
329         */
330        private void buildDotLayout() {
331                this.goToStartBtn = ComponentFactory.createImageButton("Go to start", SPH_Theme.REWIND_ICON_20PX, true);
332                if (this.exerData.isSliderEnabled()) {
333                        this.goToStartBtn.addClickListener(new Button.ClickListener() {
334
335                                private static final long serialVersionUID = 7181459362080977100L;
336
337                                @Override
338                                public void buttonClick(ClickEvent event) {
339                                        if (vjs.getCurrentTime() > 0) {
340                                                vjs.seekTo(0);
341                                        }
342                                }
343                        });
344                        this.dotLayout.addComponent(goToStartBtn);
345                }
346                if (this.exerData.isSliderEnabled()) {
347                        this.dotLayout.addComponent(this.goToStartBtn);
348                }
349                this.dotLayout.addComponents(questionDots, this.fullWindowButton);
350        }
