284        private void addListeners() {
285                // Lis‰t‰‰n videokuuntelija PAUSE-eventille.
286                videojsPlayer.addVideojsListener(Videojs.VjsListener.PAUSE_EVENT, new Videojs.VjsListener() {
287
288                        /**
289                         * 
290                         */
291                        private static final long serialVersionUID = -5423915986862325021L;
292
293                        @Override
294                        public void on(String event, Videojs source, Double time) {
295                                System.out.println( "PAUSE" );
296                                System.out.println( "Contains key: " + submissions.containsKey( nextQuestionIndex ) );
297
298                                if ( !questionButtonClicked && !submissions.containsKey( nextQuestionIndex ) &&
299                                                nextQuestionIndex < exerData.getQuestionCount() &&
300                                                time >= java.lang.Math.floor( exerData.getQuestion( nextQuestionIndex ).getTime() ) &&
301                                                time <= java.lang.Math.floor( exerData.getQuestion( nextQuestionIndex ).getTime() + 1 ) ) {
302                                        setQuestionVisible( nextQuestionIndex );
303                                        System.out.println( "Pause 2: " + nextQuestionIndex );
304                                        if ( nextQuestionIndex < exerData.getQuestionCount() ) {
305                                                System.out.println( "Next question time: " + exerData.getQuestion( nextQuestionIndex ).getTime() );
306                                        }
307                                }
308                        }
309                });
310
311                // Lis‰t‰‰n videokuuntelija PLAY-eventille.
312                videojsPlayer.addVideojsListener(Videojs.VjsListener.PLAY_EVENT, new Videojs.VjsListener() {
313
314                        /**
315                         * 
316                         */
317                        private static final long serialVersionUID = -1692620925210128858L;
318
319                        @Override
320                        public void on(String event, Videojs source, Double time) {
321                                System.out.println( "PLAY" );
322                                nextQuestionIndex = getIndexOfNextQuestionTime( time );
323                                currentQuestionPanel.setVisible( false );
324                                questionButtonClicked = false;
325                        }
326                });
327
328                // Kuuntelija continue-Buttonille.
329                continueButton.addClickListener( new Button.ClickListener() {
330
331                        /**
332                         * 
333                         */
334                        private static final long serialVersionUID = -8152740888003850116L;
335
336                        @Override
337                        public void buttonClick(ClickEvent event) {
338                                currentQuestionPanel.setVisible( false );
339                                videojsPlayer.play();
340                        }
341                });
342
343                // KUuntelija submit-Buttonille.
344                submitButton.addClickListener( new Button.ClickListener() {
345
346                        private static final long serialVersionUID = -5078227153238916321L;
347
348                        @Override
349                        public void buttonClick(ClickEvent event) {
350                                submitButton.setEnabled( false );
351                                if ( !exerData.isSliderEnabled() ) {
352                                        continueButton.setEnabled( true );
353                                        questionButtons.get( nextQuestionIndex ).getComponent( 0 ).setEnabled( false );
354                                }
355                                else {
356                                        questionButtons.get( nextQuestionIndex ).getComponent( 0 ).setEnabled( false );
357                                        questionButtons.get( nextQuestionIndex ).getComponent( 1 ).setEnabled( false );
358                                }
359                                String reply = (String) ( options.getValue() == null ? "(blank)" : options.getValue() );
360                                System.out.println( submissions.size() );
361                                //                      submissions.put( nextQuestionIndex, reply );
362                                int removeTime = exerData.getQuestion( nextQuestionIndex ).getTime();
363                                System.out.println( "Attempting to remove time " + removeTime );
364                                videojsPlayer.removePauseAtTime( removeTime );
365                                boolean correct = true;
366                                if ( reply.equals( exerData.getQuestion( nextQuestionIndex ).getCorrectAnswer(0) ) ) {
367                                        submInfo.addPoint();
368                                }
369                                else {
370                                        correct = false;
371                                }
372                                Notification n = new Notification( (correct ? "Correct" : "Incorrect"),
373                                                exerData.getQuestion( nextQuestionIndex ).getAnswerDescription() );
374                                n.setDelayMsec( -1 );
375                                n.show( Page.getCurrent() );
376                                submissions.put( nextQuestionIndex, new ArrayList< String >( Arrays.asList( reply, Boolean.toString( correct ) ) ) );
377
378                                System.out.println( submissions.keySet() );
379                                submissionArea.setReadOnly( false );
380                                submissionArea.setValue( getReplies() );
381                                submissionArea.setReadOnly( true );
382                                if ( !exerData.isSliderEnabled() ) {
383                                        videojsPlayer.unmask();
384                                }
385                        }
386                });
387
388        }

390        private void setQuestionVisible( int questionIndex ) {
391                questionLabel.setValue( "Question " + ( questionIndex + 1 ) );
392                actualQuestion.setValue( exerData.getQuestion( questionIndex ).getQuestion() );
393                options = getRadioButtons( exerData.getQuestion( questionIndex ) );
394                radioButtonsPanel.setContent( options );
395                //              nextQuestionIndex++;
396                submitButton.setEnabled( true );
397                if ( !exerData.isSliderEnabled() ) {
398                        videojsPlayer.mask();
399                        continueButton.setEnabled( false );
400                }
401                currentQuestionPanel.setVisible( true );
402        }

404        private String getReplies() {
405                String replies = "";
406                ArrayList< Integer > keyList = new ArrayList< Integer >( submissions.keySet() );
407                System.out.println( "Keylist size: " + keyList.size() );
408                for ( int i = 0; i < keyList.size(); i++ ) {
409                        replies = replies + "Question " + (keyList.get( i ) + 1) + ": " + "\n   " +
410                                        exerData.getQuestion( keyList.get( i ) ).getQuestion() + "\nAnswer:\n   " + submissions.get( keyList.get( i ) ).get( 0 ) + "   " +
411                                        (submissions.get( keyList.get( i ) ).get( 0 ).equals( exerData.getQuestion( keyList.get( i ) ).getCorrectAnswer(0) ) ? "\n(correct)" : "\n(incorrect)" ) + "\n\n";
412                }
413                return replies;
414        }

423        private int getIndexOfNextQuestionTime( double time ) {
424                int nextIndex = 0;
425                boolean indexFound = false;
426                while ( !indexFound && nextIndex < exerData.getQuestionCount() ) {
427                        //                      System.out.println( "Boolean test: " +
428                        //                                      (time < java.lang.Math.floor( exerData.getQuestion( nextIndex ).getTime() + 1)) +
429                        //                                              " && " + (!submissions.containsKey( nextIndex )) );
430                        if ( time < java.lang.Math.floor( exerData.getQuestion( nextIndex ).getTime() + 1 ) &&
431                                        !submissions.containsKey( nextIndex ) ) {
432                                indexFound = true;
433                                if ( time >= java.lang.Math.floor( exerData.getQuestion( nextIndex ).getTime() ) ) {
434                                        nextIndex++;
435                                }
436                        }
437                        else {
438                                nextIndex++;
439                        }
440                }
441                System.out.println( "returning: " + nextIndex );
442                return nextIndex;
443        }
