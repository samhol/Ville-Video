001package edu.vserver.exercises.videoMcq;
002
003import edu.vserver.exercises.model.ExerciseData;
004
005/**
006 * VideoMcqExerciseData has all the required data to create a specified Video exercise.
007 * 
008 * @author  <a href="mailto:sami.holck@gmail.com">Sami Holck</a>
009 * @author Juha Mäkilä
010 * @version v2.0
011 * @since 17.10.2013
012 * 
013 */
014public class VideoMcqExerciseData implements ExerciseData {
015
016        private static final long serialVersionUID = -716445297446246493L;
017
018        private String videoURL = "";
019        private String mimeType = "";
020        private String exerName = "";
021        private QuestionLibrary questionLibrary;
022        private boolean sliderEnabled;
023        private boolean instantResponse;
024
025        /**
026         * 
027         */
028        public VideoMcqExerciseData() {
029                this.questionLibrary = new QuestionLibrary();
030        }
031
032        /**
033         * 
034         * @param videoURL the URL of the video stream.
035         * @param mimetype the MIME type of the video stream.
036         */
037        public VideoMcqExerciseData(String videoURL, String mimetype) {
038                this(videoURL, mimetype, "", new QuestionLibrary(), false);
039        }
040
041        /**
042         * 
043         * @param videoURL the URL of the video stream.
044         * @param mimeType the MIME type of the video stream.
045         * @param name
046         * @param questions
047         * @param sliderEnabled
048         */
049        public VideoMcqExerciseData(String videoURL, String mimeType, String name, QuestionLibrary questions, boolean sliderEnabled) {
050                this.videoURL = videoURL;
051                this.mimeType = mimeType;
052                this.exerName = name;
053                this.questionLibrary = questions;
054                this.sliderEnabled = sliderEnabled;
055        }
056
057        /**
058         * Returns the URL of the video stream.
059         *
060         * @return the URL of the video stream.
061         */
062        public final String getVideoURL() {
063                return videoURL;
064        }
065
066        /**
067         * Sets the URL of the video stream.
068         *
069         * @param videoURL the URL of the video stream to set.
070         */
071        public final void setVideoURL(String videoURL) {
072                this.videoURL = videoURL;
073        }
074
075        /**
076         * Returns the MIME type of the video stream.
077         *
078         * @return the MIME type of the video stream.
079         */
080        public final String getMimeType() {
081                return mimeType;
082        }
083
084        /**
085         * Sets the MIME type of the video stream.
086         *
087         * @param mimeType the MIME type of the video stream to set.
088         */
089        public final void setMimeType(String mimeType) {
090                this.mimeType = mimeType;
091        }
092
093        /**
094         * Returns the name of the exercise.
095         *
096         * @return the name of the exercise.
097         */
098        public final String getExerName() {
099                return exerName;
100        }
101
102        /**
103         * Sets the name of the exercise.
104         *
105         * @param exerName the name of the exercise to set.
106         */
107        public final void setExerName(String exerName) {
108                this.exerName = exerName;
109        }
110
111        /**
112         * Returns the library holding all the questions.
113         *
114         * @return the library holding all the questions.
115         */
116        public final QuestionLibrary getQuestionLibrary() {
117                return questionLibrary;
118        }
119
120        /**
121         * Sets the library holding all the questions.
122         *
123         * @param questionLibrary the library holding all the questions to set.
124         */
125        public final void setQuestionLibrary(QuestionLibrary questionLibrary) {
126                this.questionLibrary = questionLibrary;
127        }
128
129        /**
130         * Indicates whether the slider is enabled in this exercise.
131         *
132         * @return true if the slider is enabled in this exercise; false otherwise.
133         */
134        public final boolean isSliderEnabled() {
135                return sliderEnabled;
136        }
137
138        /**
139         * Indicates whether the slider is enabled in this exercise.
140         *
141         * @param sliderEnabled the sliderEnabled to set
142         */
143        public final void setSliderEnabled(boolean sliderEnabled) {
144                this.sliderEnabled = sliderEnabled;
145        }
146
147        /**
148         * Specifies whether the slider is enabled in this exercise.
149         *
150         * @param sliderEnabled specifies whether the slider is enabled in this exercise.
151         */
152        public final void exerData(boolean sliderEnabled) {
153                this.sliderEnabled = sliderEnabled;
154        }
155
156        /**
157         * Indicates whether the instant response is enabled in this exercise.
158         *
159         * @return true if the instant response is enabled in this exercise; false otherwise.
160         */
161        public final boolean isInstantResponse() {
162                return instantResponse;
163        }
164
165        /**
166         * Specifies whether the instant response is enabled in this exercise.
167         *
168         * @param instantResponse the instant response is enabled (true).
169         */
170        public final void setInstantResponse(boolean instantResponse) {
171                this.instantResponse = instantResponse;
172        }
173
174        /**
175         *
176         * @return
177         * @see java.lang.Object#hashCode()
178         */
179        @Override
180        public int hashCode() {
181                final int prime = 31;
182                int result = 1;
183                result = prime * result + ((exerName == null) ? 0 : exerName.hashCode());
184                result = prime * result + (instantResponse ? 1231 : 1237);
185                result = prime * result + ((mimeType == null) ? 0 : mimeType.hashCode());
186                result = prime * result + ((questionLibrary == null) ? 0 : questionLibrary.hashCode());
187                result = prime * result + (sliderEnabled ? 1231 : 1237);
188                result = prime * result + ((videoURL == null) ? 0 : videoURL.hashCode());
189                return result;
190        }
191
192        /**
193         *
194         * @param obj
195         * @return
196         * @see java.lang.Object#equals(java.lang.Object)
197         */
198        @Override
199        public boolean equals(Object obj) {
200                if (this == obj) {
201                        return true;
202                }
203                if (obj == null) {
204                        return false;
205                }
206                if (getClass() != obj.getClass()) {
207                        return false;
208                }
209                VideoMcqExerciseData other = (VideoMcqExerciseData) obj;
210                if (exerName == null) {
211                        if (other.exerName != null) {
212                                return false;
213                        }
214                } else if (!exerName.equals(other.exerName)) {
215                        return false;
216                }
217                if (instantResponse != other.instantResponse) {
218                        return false;
219                }
220                if (mimeType == null) {
221                        if (other.mimeType != null) {
222                                return false;
223                        }
224                } else if (!mimeType.equals(other.mimeType)) {
225                        return false;
226                }
227                if (questionLibrary == null) {
228                        if (other.questionLibrary != null) {
229                                return false;
230                        }
231                } else if (!questionLibrary.equals(other.questionLibrary)) {
232                        return false;
233                }
234                if (sliderEnabled != other.sliderEnabled) {
235                        return false;
236                }
237                if (videoURL == null) {
238                        if (other.videoURL != null) {
239                                return false;
240                        }
241                } else if (!videoURL.equals(other.videoURL)) {
242                        return false;
243                }
244                return true;
245        }
246
247}
