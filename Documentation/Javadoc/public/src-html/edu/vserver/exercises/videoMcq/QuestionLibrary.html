019import java.io.Serializable;
020import java.util.ArrayList;
021import java.util.Collection;
022import java.util.Collections;
023import java.util.Iterator;
024import java.util.SortedMap;
025import java.util.TreeMap;

027/**
028 * QuestionLibrary contains all of the questions for the Video stream.
029 * 
030 * <p><strong class="Red">NOTE:</strong> The library holds no duplicates of the
031 * questions (see {@link Question#equals(Object) Question.equals(Object)}).</p>
032 * 
033 * <p><strong>copyright &copy; Ville-Video 2013</strong></p>
034 * 
035 * @.classInvariant FORALL({@link Question question} : this;
036 *                  !EXIST({@link Question q} : this;
037 *                  {@link Question#equals(Object) question.equals(q)}) && (
038 *                  {@link #getNext(Question) question.getNextQuestion()} == null ||
039 *                   question.getTime() <= {@link #getNext(Question) question.getNextQuestion()}.getTime())
040 * 
041 * @author  <a href="mailto:sami.holck@gmail.com">Sami Holck</a>
042 * @author  Juha Mäkilä
043 * @version 1.2
044 * @since   1.11.2013
045 */
046public class QuestionLibrary implements Iterable<Question>, Serializable {
047
048        private static final long serialVersionUID = 4657450323961667603L;
049
050        private final SortedMap<Integer, ArrayList<Question>> questions = Collections.synchronizedSortedMap(new TreeMap<Integer, ArrayList<Question>>());
051
052        /**
053         * Constructs an empty question library.
054         */
055        public QuestionLibrary() {
056        }
057
058        /**
059         * Constructs a question library containing a single question.
060         * 
061         * @param question new question.
062         */
063        public QuestionLibrary(Question question) {
064                this.add(question);
065        }
066
067        /**
068         * Constructs a question library containing a unique set of {@link Question questions}.
069         * 
070         * <p><strong class="Red">NOTE:</strong> The library holds no duplicates of
071         * the questions (see {@link Question#equals(Object) Question.equals(Object)}).</p>
072         * 
073         * @param questions a collection of {@link Question questions}.
074         */
075        public QuestionLibrary(Collection<Question> questions) {
076                this.addAll(questions);
077        }
078
079        /**
080         * Adds a new {@link Question question} to the library.
081         * 
082         * <p><strong class="Red">NOTE:</strong> The library holds no duplicates of
083         * the questions (see {@link Question#equals(Object) Question.equals(Object)}).</p>
084         * 
085         * @param question new question.
086         * @return true if the {@link Question question} was added to this library.
087         */
088        public boolean add(Question question) {
089                int time = question.getTime();
090                ArrayList<Question> questionList = this.questions.get(time);
091                if (questionList == null) {
092                        questionList = new ArrayList<Question>();
093                        questionList.add(question);
094                        this.questions.put(time, questionList);
095                        return true;
096                } else if (!questionList.contains(question)) {
097                        return questionList.add(question);
098                } else {
099                        return false;
100                }
101        }
102
103        /**
104         * Adds a collection of new {@link Question questions} to the library.
105         * 
106         * <p><strong class="Red">NOTE:</strong> The library holds no duplicates of
107         * the questions (see {@link Question#equals(Object) Question.equals(Object)}).</p>
108         * 
109         * @param questions new {@link Question questions}.
110         */
111        public void addAll(Collection<Question> questions) {
112                for (Question question : questions) {
113                        this.add(question);
114                }
115        }
116
117        /**
118         * Replaces the old {@link Question question} with the new one.
119         * 
120         * <p></p>
121         * 
122         * @param  oldQ the old question.
123         * @param  newQ the new question.
124         * @return true if {@link Question oldQ} was replaced to {@link Question newQ} in this library.
125         * @throws QuestionNotFoundException if the {@link Question oldQ} was not present in this library.
126         */
127        public boolean replace(Question oldQ, Question newQ) {
128                ArrayList<Question> questions = getTimeSlot(oldQ.getTime());
129                if (contains(oldQ)) {
130                        if (oldQ.getTime() == newQ.getTime()) {
131                                int index = questions.indexOf(oldQ);
132                                questions.set(index, newQ);
133                        } else {
134                                questions.remove(oldQ);
135                                this.add(newQ);
136                        }
137                        return true;
138                } else {
139                        throw new QuestionNotFoundException("Cannot replace question that is not present in the library");
140                }
141        }
142
143        /**
144         * Removes the only occurrence of the specified {@link Question question} from this library.
145         * 
146         * @param question the question to be removed from this library, if present.
147         * @return true if this library contained the specified {@link Question question}.
148         */
149        public boolean remove(Question question) {
150                int time = question.getTime();
151                ArrayList<Question> questionList = this.questions.get(time);
152                if (questionList == null) {
153                        return false;
154                }
155                boolean result = questionList.remove(question);
156                if (questionList.size() == 0) {
157                        this.questions.remove(time);
158                }
159                return result;
160        }
161
162        /**
163         * Returns the index of the first occurrence of the specified {@link Question question}
164         * in this library, or -1 if this library does not contain the {@link Question question}.
165         * 
166         * @param question question to search for.
167         * @return the index of the first occurrence of the specified {@link Question question}
168         * in this library, or -1 if this library does not contain the {@link Question question}.
169         */
170        public int indexOf(Question question) {
171                return this.getAllQuestions().indexOf(question);
172        }
173
174        /**
175         * Returns the {@link Question question} at the specified position in this library.
176         * 
177         * @param  index index of the {@link Question question} to return.
178         * @return the {@link Question question} at the specified position in this library.
179         * @throws QuestionNotFoundException if the index is out of range (index < 0 || index >= {@link #size()})
180         */
181        public Question get(int index) {
182                try {
183                        return this.getAllQuestions().get(index);
184                } catch (Exception e) {
185                        throw new QuestionNotFoundException("No question found: The index '" + index + "' is out of range");
186                }
187        }
188
189        /**
190         * Returns true if this library contains a {@link Question question} after the given one.
191         * 
192         * @param question question whose successor in this library is to be tested.
193         * @return true if this library contains a successor for the specified question.
194         */
195        public boolean containsNext(Question question) {
196                ArrayList<Question> allQuestions = this.getAllQuestions();
197                int index = allQuestions.indexOf(question);
198                return (index >= 0 && allQuestions.size() >= (index + 2));
199        }
200
201        /**
202         * Returns the next question after the the given in this library.
203         * 
204         * @param question question whose successor in this library is returned.
205         * @return the {@link Question question} at the specified position in this library or null if none was found.
206         */
207        public Question getNext(Question question) {
208                ArrayList<Question> allQuestions = this.getAllQuestions();
209                int index = allQuestions.indexOf(question);
210                if (index < 0) {
211                        return null;
212                } else if (allQuestions.size() >= (index + 2)) {
213                        return allQuestions.get(index + 1);
214                } else {
215                        return null;
216                }
217        }
218
219        /**
220         * Returns the next unanswered question after the the given in this library.
221         * 
222         * @.post  RESULT == null | !RESULT.isAnswered()
223         * @param question question whose unanswered successor in this library is returned.
224         * @return the unanswered {@link Question question} at the specified position in this library or null if none was found.
225         */
226        public Question getNextUnanswerded(Question question) {
227                Question next = this.getNext(question);
228                while (next != null && next.isAnswered()) {
229                        next = this.getNext(next);
230                }
231                return next;
232        }
233
234        /**
235         * Checks if this library contains any unanswered {@link Question questions}.
236         * 
237         * @.post  EXIST(question : this; !{@link Question#isAnswered() question.isAnswered()})
238         * @return true if this library contains any unanswered {@link Question questions}.
239         */
240        public boolean containsUnanswerded() {
241                ArrayList<Question> allQuestions = this.getAllQuestions();
242                for (Question q : allQuestions) {
243                        if (!q.isAnswered()) {
244                                return true;
245                        }
246                }
247                return false;
248        }
249
250        /**
251         * Returns the list of the {@link Question questions} at the given time slot
252         * in seconds, or null if none were found.
253         * 
254         * @.post  RESULT == null | RESULT.size() > 0
255         * @param  timeSlot the time of the fetched questions in seconds.
256         * @return the list of all the questions at the given time slot, or null.
257         */
258        public ArrayList<Question> getTimeSlot(int timeSlot) {
259                return this.questions.get(timeSlot);
260        }
261
262        /**
263         * Returns the list of all the unanswered {@link Question questions} at the
264         * given time in seconds, or null if none were found.
265         * 
266         * @.post  RESULT == null | (RESULT.size() > 0 & FORALL(q : RESULT; q.{@link Question#isAnswered() isAnswered()}))
267         * @param  timeSlot index of the {@link Question question} to return.
268         * @return the list of all the unanswered questions at the given time slot, or null.
269         */
270        public ArrayList<Question> getUnansweredFromTimeSlot(int timeSlot) {
271                if (containsTimeSlot(timeSlot)) {
272                        ArrayList<Question> theSlot = getTimeSlot(timeSlot);
273                        ArrayList<Question> unanswered = new ArrayList<Question>();
274                        for (Question q : theSlot) {
275                                if (!q.isAnswered()) {
276                                        unanswered.add(q);
277                                }
278                        }
279                        if (unanswered.size() != 0) {
280                                return unanswered;
281                        } else {
282                                return null;
283                        }
284                } else {
285                        return null;
286                }
287
288        }
289
290        /**
291         * Indicates whether there are {@link Question questions} in the given time slot and whether
292         * that time slot contains any {@link Question questions} that has not yet been answered.
293         * 
294         * @param timeSlot the time slot to be checked for a question containing any unanswered questions.
295         * @return true if at least one such a question exists; false otherwise.
296         */
297        public boolean containsUnansweredInTimeSlot(int timeSlot) {
298                return getUnansweredFromTimeSlot(timeSlot) != null;
299        }
300
301        /**
302         * Returns a sorted list of {@link Question question} time slots in this library.
303         * 
304         * @return a sorted list of {@link Question question} time slots in this library.
305         */
306        public ArrayList<Integer> getTimeSlots() {
307                ArrayList<Integer> timeSlots = new ArrayList<Integer>();
308                timeSlots.addAll(this.questions.keySet());
309                return timeSlots;
310        }
311
312        /**
313         * Checks if this library contains the specified {@link Question question}.
314         * 
315         * @param  question whose presence in this library is to be tested.
316         * @return true if this library contains the specified {@link Question question}.
317         */
318        public boolean contains(Question question) {
319                ArrayList<Question> allQuestions = this.getAllQuestions();
320                return allQuestions.contains(question);
321        }
322
323        /**
324         * Checks if this library contains any {@link Question questions} at a specified time in seconds.
325         * 
326         * @param  timeSlot the time in seconds whose presence is to be tested.
327         * @return true if this library contains any {@link Question questions} for the specified time in seconds.
328         */
329        public boolean containsTimeSlot(int timeSlot) {
330                return questions.containsKey( timeSlot );
331        }
332
333        /**
334         * Returns a containing all {@link Question questions} in the library.
335         * 
336         * @return a Collection containing all {@link Question questions} in the library.
337         */
338        public ArrayList<Question> getAllQuestions() {
339                ArrayList<Question> questions = new ArrayList<Question>();
340                for (ArrayList<Question> qlist : this.questions.values()) {
341                        questions.addAll(qlist);
342                }
343                return questions;
344        }
345
346        /**
347         * Returns the percentage of the correct given answers for all the questions
348         * in this question library.
349         * 
350         * @return the percentage of all the correct given answers.
351         */
352        public double getScore() {
353                double count = 0;
354                for (Question q : getAllQuestions()) {
355                        count += q.getQuestionScore();
356                }
357                return count / getAllQuestions().size();
358        }
359
360        /**
361         * Resets the given answers for all the questions in this question library.
362         */
363        public void resetGivenAnswers() {
364                for(Question q : this.getAllQuestions()) {
365                        q.resetGivenAnswers();
366                }
367        }
368
369        /**
370         * Returns the number of the {@link Question questions} in this library.
371         * 
372         * @return the number of the {@link Question questions} in this library.
373         */
374        public int size() {
375                int size = 0;
376                for (ArrayList<Question> qlist : this.questions.values()) {
377                        size += qlist.size();
378                }
379                return size;
380        }
381
382        /**
383         * Returns an iterator over the {@link Question} elements in this library in proper sequence.
384         * 
385         * <p>The returned iterator is fail-fast. So it throws a
386         * {@link java.util.ConcurrentModificationException ConcurrentModificationException}
387         * under either of the following two conditions.</p>
388         * 
389         * <ol>
390         * <li>In multithreaded processing: if one thread is trying to modify a
391         * <em>Collection</em> while another thread is iterating over it.</li>
392         * 
393         * <li>In single-threaded or in multithreaded processing: if after the creation
394         * of the <em>Iterator</em>, the container is modified at any time by any method
395         * other than the <em>Iterator</em>'s own <em>remove</em> or <em>add</em> methods.</em></li>
396         * </ol>
397         * @return an iterator over the {@link Question} elements in this library in proper sequence.
398         * @see java.lang.Iterable#iterator()
399         */
400        @Override
401        public Iterator<Question> iterator() {
402                return this.getAllQuestions().iterator();
403        }
404
405        /**
406         * Returns a string representation of this library.
407         *
408         * @return a string representation of this library.
409         * @see java.lang.Object#toString()
410         */
411        @Override
412        public String toString() {
413                return "QuestionLibrary [questions=" + questions + "]";
414        }
415
416        /**
417         * Returns a hash code for this question library.
418         *
419         * @return a hash code value for this object.
420         * @see    java.lang.Object#hashCode()
421         */
422        @Override
423        public int hashCode() {
424                final int prime = 31;
425                int result = 1;
426                result = prime * result + ((questions == null) ? 0 : questions.hashCode());
427                return result;
428        }

430        /**
431         * Compares this question library to the specified object.
432         *
433         * @param  obj the object to compare this against.
434         * @return true if the given object represents a question library equivalent to this, false otherwise
435         * @see    java.lang.Object#equals(java.lang.Object)
436         */
437        @Override
438        public boolean equals(Object obj) {
439                if (this == obj) {
440                        return true;
441                }
442                if (obj == null) {
443                        return false;
444                }
445                if (getClass() != obj.getClass()) {
446                        return false;
447                }
448                QuestionLibrary other = (QuestionLibrary) obj;
449                if (questions == null) {
450                        if (other.questions != null) {
451                                return false;
452                        }
453                } else if (!questions.equals(other.questions)) {
454                        return false;
455                }
456                return true;
457        }

459        /**
460         * Thrown to indicate that a {@link Question} was not found from the {@link QuestionLibrary}.
461         *
462         * @author  <a href="mailto:sami.holck@gmail.com">Sami Holck</a>
463         * @version 0.1
464         * @since   10.11.2013
465         */
466        public static class QuestionNotFoundException extends RuntimeException {
467
468                private static final long serialVersionUID = -7586358744666727024L;
469                /**
470                 * Constructs a new runtime exception with null as its detail message.
471                 * The cause is not initialized, and may subsequently be initialized by
472                 * a call to {@link Throwable#initCause(Throwable)}.
473                 */
474                public QuestionNotFoundException() {
475                        super();
476                }
477                /**
478                 * Constructs a new exception with the specified detail message.
479                 * The cause is not initialized, and may subsequently be initialized
480                 * by a call to {@link Throwable#initCause(Throwable)}.
481                 * 
482                 * @param message the detail message. The detail message is saved for
483                 *        later retrieval by the {@link Throwable#getMessage()} method.
484                 */
485                public QuestionNotFoundException(String message) {
486                        super(message);
487                }
488
489        }
