;(function ( sphSvg, undefined ) {
	"use strict";
	//var colors = ["#ccc", "#f00", "#0c0"];
	/**
	 * Constructs a new SvgDots to draw an SVG image containing small dots.
	 * @memberOf sphSvg
	 * @class
	 * @classdesc SvgDots draws an SVG image containing small dots.
	 * @constructor
	 *  
	 * @param {string} targetId the DOM id of the target element.
	 * @param {string[]} dots the source of the video.
	 * @returns {sphSvg.SvgDots} the SvgDots object.
	 */
	sphSvg.SvgDots = function (targetId, dots) {
		console.log("sphSvg.SvgDots(targetId: " + targetId + ", dots: " + dots + ")");
		/** 
		 * target id
		 * @private 
		 * */
		this.targetId = targetId;
		/** @private */
		this.target = document.getElementById(targetId);
		/** @private */
		this.dots = dots;
		/** @private */
		this.circles = [];
		/** @private */
		this.paper = Raphael(this.target, 320, 200);
		this.draw();
		this.typeOf = "sphSvg.SvgDots";
		return this;
	};

	sphSvg.SvgDots.prototype = {

		/**
		 * Does the drawing of the SVG image
		 * 
		 * @private
		 * @memberOf sphSvg.SvgDots#
		 * @returns {sphSvg.SvgDots} the SvgDots object.
		 * @param {string} targetId the DOM id of the target element
		 * @returns {sphSvg.SvgDots} the SvgDots object.
		 */
		draw : function () {
			var index = 0;
			console.log("sphSvg.SvgDots.draw()");
			for (index; index < this.dots.length; index++) {
				this.drawCircle(index);
			} 
			return this;
		}, 

		/**
		 * Draws the circle to the position of the given index
		 * 
		 * @private
		 * @memberOf sphSvg.SvgDots#
		 * @returns {sphSvg.SvgDots} the SvgDots object.
		 */
		drawCircle: function (index) {
			var that = this;
			this.circles[index] = this.paper.circle((10 + 20 * index), 14, 8);
			this.circles[index].attr({fill: this.dots[index], stroke: "black", "stroke-width": 2, title: "daa"})
				.emboss()
				.shadow();
			this.circles[index].data("index", index);
			if (this.callback) {
				this.circles[index].click(function () {
					//alert("index: " + this.data("index"));
					that.callback("click", this.data("index"));
				});	
			}
			return this;
		},

		/**
		 * Changes the color of the dot of the given index
		 * 
		 * @public
		 * @memberOf sphSvg.SvgDots#
		 * @param {int} dotNumber the index of the target dot.
		 * @param {string} dotColor the new color of the dot.
		 * @returns {sphSvg.SvgDots} the SvgDots object.
		 */
		changeDotColor: function (dotNumber, dotColor) {
			this.circles[dotNumber].attr({fill: dotColor});
		},
		
		/**
		 * 
		 * @public
		 * @memberOf sphSvg.SvgDots#
		 * @param {Function} callback the event callback function.
		 */
		setEventCallback: function (callback) {
			this.callback = callback;
		}
	};

}(window.sphSvg = window.sphSvg || {}));
